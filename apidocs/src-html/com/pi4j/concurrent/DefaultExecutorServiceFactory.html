031import java.util.ArrayList;
032import java.util.List;
033import java.util.concurrent.Executor;
034import java.util.concurrent.ExecutorService;
035import java.util.concurrent.Executors;
036import java.util.concurrent.ScheduledExecutorService;
037import java.util.concurrent.ThreadFactory;
038import java.util.concurrent.atomic.AtomicLong;
039
040public class DefaultExecutorServiceFactory implements ExecutorServiceFactory {
041
042    public static int MAX_THREADS_IN_POOL = 25;
043    private static List<ExecutorService> singleThreadExecutorServices = new ArrayList<>();
044
045    // this seemingly odd pattern is the recommended way to lazy-initialize static fields in effective java.
046    // The static "holder" class doesn't have it's static initializer called until it is accessed - and it's not accessed until the
047    // getInternalScheduledExecutor() method is called.
048    //
049    // (see effective java item 71:Use lazy initialization judiciously)
050    private static class ScheduledExecutorServiceHolder {
051        static final ScheduledExecutorService heldExecutor = Executors.newScheduledThreadPool(MAX_THREADS_IN_POOL, getThreadFactory("pi4j-scheduled-executor-%d"));
052    }
053    private static ScheduledExecutorService getInternalScheduledExecutor() {
054        return ScheduledExecutorServiceHolder.heldExecutor;
055    }
056    private static class ScheduledExecutorServiceWrapperHolder {
057        static final ScheduledExecutorServiceWrapper heldWrapper = new ScheduledExecutorServiceWrapper(getInternalScheduledExecutor());
058    }
059
060    // follow a similar lazy initialization pattern for the gpio events
061    private static ScheduledExecutorServiceWrapper getServiceWrapper() {
062        return ScheduledExecutorServiceWrapperHolder.heldWrapper;
063    }
064    private static class GpioEventServiceHolder {
065        static final ExecutorService cachedExecutor = new ShutdownDisabledExecutorWrapper(Executors.newCachedThreadPool(getThreadFactory("pi4j-gpio-event-executor-%d")));
066    }
067    private static ExecutorService getInternalGpioExecutorService() {
068        return GpioEventServiceHolder.cachedExecutor;
069    }
070
071    /**
072     * return an instance to the thread factory used to create new executor services
073     */
074    private static ThreadFactory getThreadFactory(final String nameFormat) {
075        final ThreadFactory defaultThreadFactory = Executors.defaultThreadFactory();
076        return new ThreadFactory() {
077            final AtomicLong count = (nameFormat != null) ? new AtomicLong(0) : null;
078
079            @Override
080            public Thread newThread(Runnable runnable) {
081                Thread thread = defaultThreadFactory.newThread(runnable);
082                if (nameFormat != null) {
083                    thread.setName(String.format(nameFormat, count.getAndIncrement()));
084                }
085                return thread;
086            }
087        };
088    }
089
090
091
092
093    /**
094     * return an instance to the scheduled executor service (wrapper)
095     */
096    public ScheduledExecutorService getScheduledExecutorService() {
097        // we return the protected wrapper to prevent any consumers from 
098        // being able to shutdown the scheduled executor service
099        return getServiceWrapper();
100    }
101
102    @Override
103    public ExecutorService getGpioEventExecutorService() {
104        return getInternalGpioExecutorService();
105    }
106
107    /**
108     * return a new instance of a single thread executor service
109     *
110     * This method is deprecated in favor of the getGpioEventExecutorService - which provides better guarantees around resource
111     * management
112     */
113    @Override
114    public ExecutorService newSingleThreadExecutorService() {
115       return Executors.newSingleThreadExecutor(getThreadFactory("pi4j-single-executor-%d"));
116    }
117
118    /**
119     * shutdown executor threads
120     */
121    public void shutdown() {
122        // shutdown each single thread executor in the managed collection
123        for (ExecutorService singleThreadExecutorService : singleThreadExecutorServices) {
124            shutdownExecutor(singleThreadExecutorService);
125        }
126
127        // shutdown scheduled executor instance
128        shutdownExecutor(getInternalScheduledExecutor());
129        shutdownExecutor(getInternalGpioExecutorService());
130
131    }
132
133    private void shutdownExecutor(ExecutorService executor) {
134        if (executor != null) {
135            if (!executor.isShutdown()) {
136                // this is a forceful shutdown;
137                // don't wait for the scheduled tasks to complete
138                executor.shutdownNow();
139            }
140        }
141    }
142}
