031import java.util.Collection;

035public interface GpioController

040    void unexportAll();
041
042    void setMode(PinMode mode, GpioPin... pin);

044    boolean isMode(PinMode mode, GpioPin... pin);
045
046    void setPullResistance(PinPullResistance resistance, GpioPin... pin);
047    PinPullResistance getPullResistance(GpioPin pin);
048    boolean isPullResistance(PinPullResistance resistance, GpioPin... pin);
049
050    void high(GpioPinDigitalOutput... pin);
051    boolean isHigh(GpioPinDigital... pin);
052
053    void low(GpioPinDigitalOutput... pin);
054    boolean isLow(GpioPinDigital... pin);
055    
056    void setState(PinState state, GpioPinDigitalOutput... pin);
057    void setState(boolean state, GpioPinDigitalOutput... pin);
058    boolean isState(PinState state, GpioPinDigital... pin);

061    void toggle(GpioPinDigitalOutput... pin);

080    GpioPinDigitalInput provisionDigitalInputPin(GpioProvider provider, Pin pin, String name);

082    GpioPinDigitalInput provisionDigitalInputPin(Pin pin, String name);

084    GpioPinDigitalOutput provisionDigitalOutputPin(GpioProvider provider, Pin pin, String name, PinState defaultState);
085    GpioPinDigitalOutput provisionDigitalOutputPin(GpioProvider provider, Pin pin, String name);
086    GpioPinDigitalOutput provisionDigitalOutputPin(Pin pin, String name, PinState defaultState);
087    GpioPinDigitalOutput provisionDigitalOutputPin(Pin pin, String name);
088
089    GpioPinAnalogInput provisionAnalogInputPin(GpioProvider provider, Pin pin, String name);

091    
092    GpioPinAnalogOutput provisionAnalogOutputPin(GpioProvider provider, Pin pin, String name, double defaultValue);
093    GpioPinAnalogOutput provisionAnalogOutputPin(GpioProvider provider, Pin pin, String name);
094    GpioPinAnalogOutput provisionAnalogOutputPin(Pin pin, String name, double defaultValue);
095    GpioPinAnalogOutput provisionAnalogOutputPin(Pin pin, String name);
096
097    GpioPinPwmOutput provisionPwmOutputPin(GpioProvider provider, Pin pin, String name, int defaultValue);
098    GpioPinPwmOutput provisionPwmOutputPin(GpioProvider provider, Pin pin, String name);
099    GpioPinPwmOutput provisionPwmOutputPin(Pin pin, String name, int defaultValue);
100    GpioPinPwmOutput provisionPwmOutputPin(Pin pin, String name);
101    
102    GpioPin provisionPin(GpioProvider provider, Pin pin, String name, PinMode mode);
103    GpioPin provisionPin(Pin pin, String name, PinMode mode);
104
105    Collection<GpioPin> getProvisionedPins();
106    
107    void unprovisionPin(GpioPin... pin);
108}
