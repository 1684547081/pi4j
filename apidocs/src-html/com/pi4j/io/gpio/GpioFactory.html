019import com.pi4j.io.gpio.impl.GpioController;

021/**
022 * <h1>GPIO Factory</h1>
023 * 
024 * <p>
025 * This factory class provides a static method to create new 'Gpio' instances.
026 * </p>
027 * 
028 * <p>
029 * Before using the Pi4J library, you need to ensure that the Java VM in configured with access to
030 * the following system libraries:
031 * <ul>
032 * <li>pi4j</li>
033 * <li>wiringPi</li>
034 * </ul>
035 * <blockquote> This library depends on the wiringPi native system library.</br> (developed by
036 * Gordon Henderson @ <a href="https://projects.drogon.net/">https://projects.drogon.net/</a>)
037 * </blockquote>
038 * </p>
039 * 
040 * @see #com.pi4j.io.gpio.Gpio
041 * 
042 * @see <a href="http://www.pi4j.com/">http://www.pi4j.com/</a>
043 * @author Robert Savage (<a
044 *         href="http://www.savagehomeautomation.com">http://www.savagehomeautomation.com</a>)
045 */
046public class GpioFactory
047{
048    /**
049     * <h1>Create New Serial instance</h1>
050     * 
051     * @return <p>
052     *         Return a new Serial impl instance.
053     *         </p>
054     */
055    public static Gpio createInstance()
056    {
057        return new GpioController();
058    }
059}
