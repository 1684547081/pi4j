001package com.pi4j.io.gpio;
002
003public enum GpioPin
004{
005    GPIO_00(0, "GPIO 0", "SDA"), // HEADER PIN 03
006    GPIO_01(1, "GPIO 1", "SCL"), // HEADER PIN 05
007    GPIO_04(4, "GPIO 4", "GPCLK0"), // HEADER PIN 07
008    GPIO_07(7, "GPIO 7", "CE1"), // HEADER PIN 26
009    GPIO_08(8, "GPIO 8", "CE0"), // HEADER PIN 24
010    GPIO_09(9, "GPIO 9", "MISO"), // HEADER PIN 21
011    GPIO_10(10, "GPIO 10", "MOSI"), // HEADER PIN 19
012    GPIO_11(11, "GPIO 11", "SCKL"), // HEADER PIN 23
013    GPIO_14(14, "GPIO 14", "TXD"), // HEADER PIN 08
014    GPIO_15(15, "GPIO 15", "RXD"), // HEADER PIN 10
015    GPIO_17(17, "GPIO 17"), // HEADER PIN 11
016    GPIO_18(18, "GPIO 18", "PCM_CLK"), // HEADER PIN 12
017    GPIO_21(21, "GPIO 21", "PCM_DOUT"), // HEADER PIN 13
018    GPIO_22(22, "GPIO 22"), // HEADER PIN 15
019    GPIO_23(23, "GPIO 23"), // HEADER PIN 16
020    GPIO_24(24, "GPIO 24"), // HEADER PIN 18
021    GPIO_25(25, "GPIO 25"); // HEADER PIN 22
022
023    private int value;
024    private String name = null;
025    private String altFunction = null;
026
027    private GpioPin(int value, String name, String altFunction)
028    {
029        this.value = value;
030        this.name = name;
031        this.altFunction = altFunction;
032    }
033
034    private GpioPin(int value, String name)
035    {
036        this.value = value;
037        this.name = name;
038    }
039
040    public int getValue()
041    {
042        return value;
043    }
044
045    public String getValueString()
046    {
047        return Integer.toString(value);
048    }
049    
050    public String getName()
051    {
052        return name;
053    }
054
055    public String getAltFunction()
056    {
057        return altFunction;
058    }
059
060    public boolean hasAltFunction()
061    {
062        return (altFunction != null && !altFunction.isEmpty());
063    }
064
065    @Override
066    public String toString()
067    {
068        if (hasAltFunction())
069            return name + " (" + altFunction + ")";
070        else
071            return name;
072    }
073
074    public static GpioPin[] allPins()
075    {
076        return GpioPin.values();
077    }
078
079    public static GpioPin getPin(int pinNumber)
080    {
081        for (GpioPin pin : GpioPin.values())
082        {
083            if (pin.getValue() == pinNumber)
084                return pin;
085        }
086        return null;
087    }
088
089    public static GpioPin getPin(String pinName)
090    {
091        for (GpioPin pin : GpioPin.values())
092        {
093            if (pin.getValueString().equalsIgnoreCase(pinName.trim()))
094                return pin;
095
096            else if (pin.getName().equalsIgnoreCase(pinName.trim()))
097                return pin;
098            
099            else if (pin.hasAltFunction() && pin.getAltFunction().equalsIgnoreCase(pinName.trim()))
100                return pin;
101        }
102        return null;
103    }
104}
