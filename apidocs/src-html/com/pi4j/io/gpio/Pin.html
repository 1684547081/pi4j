031public enum Pin
032{
033    GPIO_00(0, "GPIO 0"),             // P1 HEADER PIN 11
034    GPIO_01(1, "GPIO 1","PCM_CLK"),   // P1 HEADER PIN 12
035    GPIO_02(2, "GPIO 2", "PCM_DOUT"), // P1 HEADER PIN 13
036    GPIO_03(3, "GPIO 3"),             // P1 HEADER PIN 15
037    GPIO_04(4, "GPIO 4"),             // P1 HEADER PIN 16
038    GPIO_05(5, "GPIO 5"),             // P1 HEADER PIN 18
039    GPIO_06(6, "GPIO 6"),             // P1 HEADER PIN 22
040    GPIO_07(7, "GPIO 7", "GPCLK0"),   // P1 HEADER PIN 07
041    GPIO_08(8, "GPIO 8", "SDA0"),     // P1 HEADER PIN 03
042    GPIO_09(9, "GPIO 9", "SCL0"),     // P1 HEADER PIN 05
043    GPIO_10(10, "GPIO 10", "CE0"),    // P1 HEADER PIN 24
044    GPIO_11(11, "GPIO 11", "CE1"),    // P1 HEADER PIN 26
045    GPIO_12(12, "GPIO 12", "MOSI"),   // P1 HEADER PIN 19
046    GPIO_13(13, "GPIO 13", "MISO"),   // P1 HEADER PIN 21
047    GPIO_14(14, "GPIO 14", "SCLK"),   // P1 HEADER PIN 23
048    GPIO_15(15, "GPIO 15", "TxD"),    // P1 HEADER PIN 08
049    GPIO_16(16, "GPIO 16", "RxD"),    // P1 HEADER PIN 10
050    GPIO_17(17, "GPIO 17"),           // P5 HEADER PIN 3
051    GPIO_18(18, "GPIO 18"),           // P5 HEADER PIN 4 
052    GPIO_19(19, "GPIO 19"),           // P5 HEADER PIN 5
053    GPIO_20(20, "GPIO 20");           // P5 HEADER PIN 6
054
055    private int value;
056    private String name = null;
057    private String altFunction = null;
058
059    private Pin(int value, String name, String altFunction)
060    {
061        this.value = value;
062        this.name = name;
063        this.altFunction = altFunction;
064    }
065
066    private Pin(int value, String name)
067    {
068        this.value = value;
069        this.name = name;
070    }
071
072    public int getValue()
073    {
074        return value;
075    }
076
077    public String getValueString()
078    {
079        return Integer.toString(value);
080    }
081    
082    public String getName()
083    {
084        return name;
085    }
086
087    public String getAltFunction()
088    {
089        return altFunction;
090    }
091
092    public boolean hasAltFunction()
093    {
094        return (altFunction != null && !altFunction.isEmpty());
095    }
096
097    @Override
098    public String toString()
099    {
100        if (hasAltFunction())
101            return name + " (" + altFunction + ")";
102        else
103            return name;
104    }
105
106    public static Pin[] allPins()
107    {
108        return Pin.values();
109    }
110
111    public static Pin getPin(int pinNumber)
112    {
113        for (Pin pin : Pin.values())
114        {
115            if (pin.getValue() == pinNumber)
116                return pin;
117        }
118        return null;
119    }
120
121    public static Pin getPin(String pinName)
122    {
123        for (Pin pin : Pin.values())
124        {
125            if (pin.getValueString().equalsIgnoreCase(pinName.trim()))
126                return pin;
127
128            else if (pin.getName().equalsIgnoreCase(pinName.trim()))
129                return pin;
130            
131            else if (pin.hasAltFunction() && pin.getAltFunction().equalsIgnoreCase(pinName.trim()))
132                return pin;
133        }
134        return null;
135    }
136}
