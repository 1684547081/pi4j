031public enum Pin
032{
033    GPIO_00(0, "GPIO 0", "SDA"), // HEADER PIN 03
034    GPIO_01(1, "GPIO 1", "SCL"), // HEADER PIN 05
035    GPIO_04(4, "GPIO 4", "GPCLK0"), // HEADER PIN 07
036    GPIO_07(7, "GPIO 7", "CE1"), // HEADER PIN 26
037    GPIO_08(8, "GPIO 8", "CE0"), // HEADER PIN 24
038    GPIO_09(9, "GPIO 9", "MISO"), // HEADER PIN 21
039    GPIO_10(10, "GPIO 10", "MOSI"), // HEADER PIN 19
040    GPIO_11(11, "GPIO 11", "SCKL"), // HEADER PIN 23
041    GPIO_14(14, "GPIO 14", "TXD"), // HEADER PIN 08
042    GPIO_15(15, "GPIO 15", "RXD"), // HEADER PIN 10
043    GPIO_17(17, "GPIO 17"), // HEADER PIN 11
044    GPIO_18(18, "GPIO 18", "PCM_CLK"), // HEADER PIN 12
045    GPIO_21(21, "GPIO 21", "PCM_DOUT"), // HEADER PIN 13
046    GPIO_22(22, "GPIO 22"), // HEADER PIN 15
047    GPIO_23(23, "GPIO 23"), // HEADER PIN 16
048    GPIO_24(24, "GPIO 24"), // HEADER PIN 18
049    GPIO_25(25, "GPIO 25"); // HEADER PIN 22
050
051    private int value;
052    private String name = null;
053    private String altFunction = null;
054
055    private Pin(int value, String name, String altFunction)
056    {
057        this.value = value;
058        this.name = name;
059        this.altFunction = altFunction;
060    }
061
062    private Pin(int value, String name)
063    {
064        this.value = value;
065        this.name = name;
066    }
067
068    public int getValue()
069    {
070        return value;
071    }
072
073    public String getValueString()
074    {
075        return Integer.toString(value);
076    }
077    
078    public String getName()
079    {
080        return name;
081    }
082
083    public String getAltFunction()
084    {
085        return altFunction;
086    }
087
088    public boolean hasAltFunction()
089    {
090        return (altFunction != null && !altFunction.isEmpty());
091    }
092
093    @Override
094    public String toString()
095    {
096        if (hasAltFunction())
097            return name + " (" + altFunction + ")";
098        else
099            return name;
100    }
101
102    public static Pin[] allPins()
103    {
104        return Pin.values();
105    }
106
107    public static Pin getPin(int pinNumber)
108    {
109        for (Pin pin : Pin.values())
110        {
111            if (pin.getValue() == pinNumber)
112                return pin;
113        }
114        return null;
115    }
116
117    public static Pin getPin(String pinName)
118    {
119        for (Pin pin : Pin.values())
120        {
121            if (pin.getValueString().equalsIgnoreCase(pinName.trim()))
122                return pin;
123
124            else if (pin.getName().equalsIgnoreCase(pinName.trim()))
125                return pin;
126            
127            else if (pin.hasAltFunction() && pin.getAltFunction().equalsIgnoreCase(pinName.trim()))
128                return pin;
129        }
130        return null;
131    }
132}
