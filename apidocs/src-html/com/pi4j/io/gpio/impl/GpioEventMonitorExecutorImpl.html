029import com.pi4j.io.gpio.GpioFactory;
030import com.pi4j.io.gpio.GpioPinDigitalInput;
031import com.pi4j.io.gpio.GpioPinInput;
032import com.pi4j.io.gpio.PinState;
033import com.pi4j.io.gpio.event.PinDigitalStateChangeEvent;
034import com.pi4j.io.gpio.event.PinEvent;
035import com.pi4j.io.gpio.event.PinEventType;
036import com.pi4j.io.gpio.event.PinListener;
037import com.pi4j.io.gpio.tasks.impl.GpioEventDebounceTaskImpl;
038import com.pi4j.io.gpio.tasks.impl.GpioEventDispatchTaskImpl;
039
040import java.util.concurrent.ExecutorService;
041import java.util.concurrent.ScheduledExecutorService;
042import java.util.concurrent.ScheduledFuture;
043import java.util.concurrent.TimeUnit;
044
045public class GpioEventMonitorExecutorImpl implements PinListener {
046    
047    private final GpioPinInput pin;
048    private static ExecutorService executor;
049    private static ScheduledExecutorService scheduledExecutor;
050    private ScheduledFuture debounceFuture = null;
051    
052    public GpioEventMonitorExecutorImpl(GpioPinInput pin) {
053        this.pin = pin;        
054        executor = GpioFactory.getExecutorServiceFactory().getGpioEventExecutorService();
055        scheduledExecutor = GpioFactory.getExecutorServiceFactory().getScheduledExecutorService();
056    }
057    
058    @Override
059    public void handlePinEvent(PinEvent event) {
060
061        // for digital input pins, we need to enforce pin debounce event suppression
062        if(pin instanceof GpioPinDigitalInput && event.getEventType() == PinEventType.DIGITAL_STATE_CHANGE){
063
064            // cast to the digital input pin interface, get the current state from
065            // the event, and determine the debounce interval for this pin state
066            GpioPinDigitalInput dip = (GpioPinDigitalInput)pin;
067            PinState state = ((PinDigitalStateChangeEvent) event).getState();
068            int pinDebounceForState = dip.getDebounce(state);
069
070            // if the pin has a debounce delay configured for this pin state,
071            // then we will need to use the pin debounce logic to defer pin events
072            if(pinDebounceForState > 0) {
073                // if no existing debounce future task exists or no future task is still busy, then
074                // create a new debounce future task and schedule it based on the debounce
075                // interval defined for the current pin state
076                if (debounceFuture == null || debounceFuture.isDone()) {
077                    debounceFuture = scheduledExecutor.schedule(
078                            new GpioEventDebounceTaskImpl(dip, state), pinDebounceForState, TimeUnit.MILLISECONDS);
079                } else {
080                    // if an existing debounce future task exists and is still busy,
081                    // then exit this method effectively suppressing the current pin event
082                    return;
083                }
084            }
085        }
086
087        // add a new pin event notification to the thread pool for *immediate* execution
088        executor.execute(new GpioEventDispatchTaskImpl(pin, event));
089    }
090}
