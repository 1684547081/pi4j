001package com.pi4j.io.gpio.impl;
002
003import java.util.Map.Entry;
004import java.util.concurrent.Callable;
005import java.util.concurrent.Executors;
006import java.util.concurrent.ScheduledExecutorService;
007import java.util.concurrent.ScheduledFuture;
008import java.util.concurrent.TimeUnit;
009import java.util.concurrent.ConcurrentHashMap;
010
011import com.pi4j.io.gpio.Gpio;
012import com.pi4j.io.gpio.GpioPin;
013import com.pi4j.io.gpio.GpioPinState;
014
015public class GpioPulse 
016{
017    @SuppressWarnings("rawtypes")
018    private static ConcurrentHashMap<GpioPin, ScheduledFuture> tasks = new ConcurrentHashMap<GpioPin, ScheduledFuture>();
019    private static ScheduledExecutorService scheduledExecutorService = Executors.newScheduledThreadPool(5);
020
021    public synchronized static void execute(Gpio gpio, GpioPin pin, long milliseconds)
022    {        
023        // first determine if an existing pulse job is already scheduled for this pin
024        if(tasks.containsKey(pin))
025        {
026            // if a job is found, then cancel it immediately and remove the job
027            ScheduledFuture<?> previouslyScheduled= tasks.get(pin);
028            previouslyScheduled.cancel(true);            
029            tasks.remove(pin);
030        }
031
032        // set the high state
033        gpio.setState(pin, GpioPinState.HIGH);
034
035        // create future job to return the pin to the low state
036        ScheduledFuture<?> scheduledFuture =        
037            scheduledExecutorService.schedule(new GpioPulseOff(gpio, pin), milliseconds, TimeUnit.MILLISECONDS);
038        
039        // add the new scheduled job to the tasks map
040        tasks.put(pin, scheduledFuture);
041        
042        // create future job to return the pin to the low state
043        @SuppressWarnings({ "rawtypes", "unchecked", "unused" })
044        ScheduledFuture<?> cleanupFuture =        
045                scheduledExecutorService.schedule(new Callable() 
046                {
047                    public Object call() throws Exception 
048                    {
049                        for(Entry<GpioPin, ScheduledFuture> item : tasks.entrySet())
050                        {   
051                            ScheduledFuture task = item.getValue(); 
052                            if(task.isCancelled() || task.isDone())
053                                tasks.remove(item.getKey());
054                        }
055                        return null;
056                    }
057                },
058                (milliseconds  + 500),
059                TimeUnit.MILLISECONDS);   
060    }     
061}
