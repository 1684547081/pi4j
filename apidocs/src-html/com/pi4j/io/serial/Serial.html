017package com.pi4j.io.serial;
018
019/**
020 * <h1>Serial Interface</h1>
021 * 
022 * <p>
023 * This interface provides a set of functions for 'Serial' communication.
024 * </p>
025 * 
026 * <p>
027 * Before using the Pi4J library, you need to ensure that the Java VM in configured with access to
028 * the following system libraries:
029 * <ul>
030 * <li>pi4j</li>
031 * <li>wiringPi</li>
032 * </ul>
033 * <blockquote> This library depends on the wiringPi native system library.</br> (developed by
034 * Gordon Henderson @ <a href="https://projects.drogon.net/">https://projects.drogon.net/</a>)
035 * </blockquote>
036 * </p>
037 * 
038 * @see #SerialFactory
039 * @see #SerialDataEvent
040 * @see #SerialDataListener
041 * 
042 * @see <a href="http://www.pi4j.com/">http://www.pi4j.com/</a>
043 * @author Robert Savage (<a
044 *         href="http://www.savagehomeautomation.com">http://www.savagehomeautomation.com</a>)
045 */
046public interface Serial
047{
048    /**
049     * The default hardware COM port provided via the Raspberry Pi GPIO header.
050     * 
051     * @see #serialOpen(String,int)
052     */
053    public static final String DEFAULT_COM_PORT = "/dev/ttyAMA0";
054    
055    
056    int open(String device, int baudRate);    
057    
058    /**
059     * Close the serial port.
060     */
061    void close();
062    
063    boolean isOpen();
064    void flush();
065    char read();
066    char readLine();
067    void write(char data);
068    void write(char data[]);
069    void write(byte data);
070    void write(byte data[]);
071    void write(String data);
072    void write(String data, String...args);
073    int availableBytes();
074    void addListener(SerialDataListener listener);
075    void removeListener(SerialDataListener listener);
076}
