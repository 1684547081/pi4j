031/**
032 * <p>This interface provides a set of functions for 'Serial' communication.</p>
033 * 
034 * <p>
035 * Before using the Pi4J library, you need to ensure that the Java VM in configured with access to
036 * the following system libraries:
037 * <ul>
038 * <li>pi4j</li>
039 * <li>wiringPi</li>
040 * </ul>
041 * <blockquote> This library depends on the wiringPi native system library.</br> (developed by
042 * Gordon Henderson @ <a href="http://wiringpi.com/">http://wiringpi.com/</a>)
043 * </blockquote>
044 * </p>
045 * 
046 * @see com.pi4j.io.serial.SerialFactory
047 * @see com.pi4j.io.serial.SerialDataEvent
048 * @see com.pi4j.io.serial.SerialDataListener
049 * 
050 * @see <a href="http://www.pi4j.com/">http://www.pi4j.com/</a>
051 * @author Robert Savage (<a
052 *         href="http://www.savagehomeautomation.com">http://www.savagehomeautomation.com</a>)
053 */
054@SuppressWarnings("unused")
055public interface Serial {
056
057    /**
058     * The default hardware COM port provided via the Raspberry Pi GPIO header.
059     * 
060     * @see #open(String, int)
061     */
062    public static final String DEFAULT_COM_PORT = "/dev/ttyAMA0";
063    public static final int DEFAULT_MONITOR_INTERVAL = 100; // milliseconds
064
065    /**
066     * This method is call to open a serial port for communication.
067     * 
068     * @see #DEFAULT_COM_PORT
069     * 
070     * @param device The device address of the serial port to access. You can use constant
071     *            'DEFAULT_COM_PORT' if you wish to access the default serial port provided via the
072     *            GPIO header.
073     * @param baudRate The baud rate to use with the serial port.
074     *            
075     * @throws SerialPortException Exception thrown on any error.
076     */
077    public void open(String device, int baudRate) throws SerialPortException;
078
079    /**
080     * This method is called to close a currently open open serial port.
081     */
082    public void close() throws IllegalStateException;
083
084    /**
085     * This method is called to determine if the serial port is already open.
086     * 
087     * @see #open(String, int)
088     * @return a value of 'true' is returned if the serial port is already open.
089     */
090    public boolean isOpen();
091
092    /**
093     * This method is called to determine if the serial port is already closed.
094     * 
095     * @see #open(String, int)
096     * @return a value of 'true' is returned if the serial port is already in the closed state.
097     */
098    public boolean isClosed();
099    
100    /**
101     * This method is called to immediately flush the serial data transmit buffer and force any
102     * pending data to be sent to the serial port immediately.
103     */
104    public void flush() throws IllegalStateException;
105
106    /**
107     * <p>This method will read the next character available from the serial port receive buffer.</p>
108     * <p>
109     * <b>NOTE: If a serial data listener has been implemented and registered with this class, then
110     * this method should not be called directly. A background thread will be running to collect
111     * received data from the serial port receive buffer and the received data will be available on
112     * via the event.</b>
113     * </p>
114     * 
115     * @return next available character in the serial data buffer
116     */
117    public char read() throws IllegalStateException;
118
119    /**
120     * This method is called to submit a single character of data to the serial port transmit
121     * buffer.
122     * 
123     * @param data A single character to be transmitted.
124     */
125    public void write(char data) throws IllegalStateException;
126
127    /**
128     * This method is called to submit a character array of data to the serial port transmit buffer.
129     * 
130     * @param data  A character array of data to be transmitted.
131     */
132    public void write(char data[]) throws IllegalStateException;
133
134    /**
135     * This method is called to submit a single byte of data to the serial port transmit buffer.
136     * 
137     * @param data  A single byte to be transmitted.
138     */
139    public void write(byte data) throws IllegalStateException;
140
141    /**
142     * This method is called to submit a byte array of data to the serial port transmit buffer.
143     * 
144     * @param data A byte array of data to be transmitted.
145     */
146    public void write(byte data[]) throws IllegalStateException;
147
148    /**
149     * This method is called to submit a string of data to the serial port transmit buffer.
150     * 
151     * @param data  A string of data to be transmitted.
152     */
153    public void write(String data) throws IllegalStateException;
154
155    /**
156     * This method is called to submit a string of data with trailing CR + LF characters to the
157     * serial port transmit buffer.
158     * 
159     * @param data  A string of data to be transmitted.
160     */
161    public void writeln(String data) throws IllegalStateException;
162
163    /**
164     * This method is called to submit a string of formatted data to the serial port transmit
165     * buffer.
166     * 
167     * @param data  A string of formatted data to be transmitted.
168     * @param args  A series of arguments that can be included for the format string variable
169     *            replacements.
170     */
171    public void write(String data, String... args) throws IllegalStateException;
172
173    /**
174     * This method is called to submit a string of formatted data with trailing CR + LF characters
175     * to the serial port transmit buffer.
176     * 
177     * @param data  A string of formatted data to be transmitted.
178     * @param args  A series of arguments that can be included for the format string variable
179     *            replacements.
180     */
181    public void writeln(String data, String... args) throws IllegalStateException;
182
183    /**
184     * This method is called to determine if and how many bytes are available on the serial received
185     * data buffer.
186     * 
187     * @return The number of available bytes pending in the serial received buffer is returned.
188     */
189    public int availableBytes() throws IllegalStateException;
190
191    /**
192     * <p>
193     * Java consumer code can call this method to register itself as a listener for serial data
194     * events.
195     * </p>
196     * 
197     * @see com.pi4j.io.serial.SerialDataListener
198     * @see com.pi4j.io.serial.SerialDataEvent
199     * 
200     * @param listener  A class instance that implements the SerialListener interface.
201     */
202    public void addListener(SerialDataListener... listener);
203
204    /**
205     * <p> Java consumer code can call this method to unregister itself as a listener for serial data
206     * events. </p>
207     * 
208     * @see com.pi4j.io.serial.SerialDataListener
209     * @see com.pi4j.io.serial.SerialDataEvent
210     * 
211     * @param listener A class instance that implements the SerialListener interface.
212     */
213    public void removeListener(SerialDataListener... listener);
214        
215    /**
216     * This method returns TRUE if the serial interface has been shutdown.
217     * 
218     * @return shutdown state
219     */
220    public boolean isShutdown();
221    
222    
223    /**
224     * This method can be called to forcefully shutdown all serial data monitoring threads.
225     */
226    public void shutdown();
227
228    /**
229     * This method returns the serial data receive monitor delay interval in milliseconds.
230     * @return interval milliseconds
231     */
232    public int getMonitorInterval();
233
234    /**
235     * This method set the serial data receive monitor delay interval in milliseconds.
236     *
237     * @param interval number of milliseconds
238     */
239    public void setMonitorInterval(int interval);
240}
