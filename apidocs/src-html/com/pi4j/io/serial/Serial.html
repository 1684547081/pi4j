031/**
032 * <h1>Serial Interface</h1>
033 * 
034 * <p>
035 * This interface provides a set of functions for 'Serial' communication.
036 * </p>
037 * 
038 * <p>
039 * Before using the Pi4J library, you need to ensure that the Java VM in configured with access to
040 * the following system libraries:
041 * <ul>
042 * <li>pi4j</li>
043 * <li>wiringPi</li>
044 * </ul>
045 * <blockquote> This library depends on the wiringPi native system library.</br> (developed by
046 * Gordon Henderson @ <a href="https://projects.drogon.net/">https://projects.drogon.net/</a>)
047 * </blockquote>
048 * </p>
049 * 
050 * @see #SerialFactory
051 * @see #SerialDataEvent
052 * @see #SerialDataListener
053 * 
054 * @see <a href="http://www.pi4j.com/">http://www.pi4j.com/</a>
055 * @author Robert Savage (<a
056 *         href="http://www.savagehomeautomation.com">http://www.savagehomeautomation.com</a>)
057 */
058public interface Serial
059{
060    /**
061     * <p>
062     * The default hardware COM port provided via the Raspberry Pi GPIO header.
063     * </p>
064     * 
065     * @see #serialOpen(String,int)
066     */
067    public static final String DEFAULT_COM_PORT = "/dev/ttyAMA0";
068
069    /**
070     * <p>
071     * This method is call to open a serial port for communication.
072     * </p>
073     * 
074     * @see #DEFAULT_COM_PORT
075     * 
076     * @param device <p>
077     *            <The device address of the serial port to access. You can use constant
078     *            'DEFAULT_COM_PORT' if you wish to access the default serial port provided via the
079     *            GPIO header./p>
080     * @param baudRate <p>
081     *            The baud rate to use with the serial port.
082     *            </p>
083     * @return <p>
084     *         The return value is the file descriptor or -1 for any error.
085     *         </p>
086     */
087    int open(String device, int baudRate);
088
089    /**
090     * <p>
091     * This method is called to close a currently open open serial port.
092     * </p>
093     */
094    void close();
095
096    /**
097     * <p>
098     * This method is called to determine if the serial port is already open.
099     * </p>
100     * 
101     * @see #open(String, int)
102     * @return <p>
103     *         a value of 'true' is returned if the serial port is already open.
104     *         </p>
105     */
106    boolean isOpen();
107
108    /**
109     * <p>
110     * This method is called to immediately flush the serial data transmit buffer and force any
111     * pending data to be sent to the serial port immediately.
112     * </p>
113     */
114    void flush();
115
116    /**
117     * <p>
118     * This method will read the next character available from the serial port receive buffer.
119     * </p>
120     * <p>
121     * <b>NOTE: If a serial data listener has been implemented and registered with this class, then
122     * this method should not be called directly. A background thread will be running to collect
123     * received data from the serial port receive buffer and the received data will be available on
124     * via the event.</b>
125     * </p>
126     * 
127     * @return <p>
128     *         next available character in the serial data buffer
129     *         </p>
130     */
131    char read();
132
133    /**
134     * <p>
135     * This method is called to submit a single character of data to the serial port transmit
136     * buffer.
137     * </p>
138     * 
139     * @param data <p>
140     *            A single character to be transmitted.
141     *            </p>
142     */
143    void write(char data);
144
145    /**
146     * <p>
147     * This method is called to submit a character array of data to the serial port transmit buffer.
148     * </p>
149     * 
150     * @param data <p>
151     *            A character array of data to be transmitted.
152     *            </p>
153     */
154    void write(char data[]);
155
156    /**
157     * <p>
158     * This method is called to submit a single byte of data to the serial port transmit buffer.
159     * </p>
160     * 
161     * @param data <p>
162     *            A single byte to be transmitted.
163     *            </p>
164     */
165    void write(byte data);
166
167    /**
168     * <p>
169     * This method is called to submit a byte array of data to the serial port transmit buffer.
170     * </p>
171     * 
172     * @param data <p>
173     *            A byte array of data to be transmitted.
174     *            </p>
175     */
176    void write(byte data[]);
177
178    /**
179     * <p>
180     * This method is called to submit a string of data to the serial port transmit buffer.
181     * </p>
182     * 
183     * @param data <p>
184     *            A string of data to be transmitted.
185     *            </p>
186     */
187    void write(String data);
188
189    /**
190     * <p>
191     * This method is called to submit a string of data with trailing CR + LF characters to the
192     * serial port transmit buffer.
193     * </p>
194     * 
195     * @param data <p>
196     *            A string of data to be transmitted.
197     *            </p>
198     */
199    void writeln(String data);
200
201    /**
202     * <p>
203     * This method is called to submit a string of formatted data to the serial port transmit
204     * buffer.
205     * </p>
206     * 
207     * @param data <p>
208     *            A string of formatted data to be transmitted.
209     *            </p>
210     * @param args <p>
211     *            A series of arguments that can be included for the format string variable
212     *            replacements.
213     *            </p>
214     */
215    void write(String data, String... args);
216
217    /**
218     * <p>
219     * This method is called to submit a string of formatted data with trailing CR + LF characters
220     * to the serial port transmit buffer.
221     * </p>
222     * 
223     * @param data <p>
224     *            A string of formatted data to be transmitted.
225     *            </p>
226     * @param args <p>
227     *            A series of arguments that can be included for the format string variable
228     *            replacements.
229     *            </p>
230     */
231    void writeln(String data, String... args);
232
233    /**
234     * <p>
235     * This method is called to determine if and how many bytes are available on the serial received
236     * data buffer.
237     * </p>
238     * 
239     * @return <p>
240     *         The number of available bytes pending in the serial received buffer is returned.
241     *         </p>
242     */
243    int availableBytes();
244
245    /**
246     * <h1>Add Serial Event Listener</h1>
247     * 
248     * <p>
249     * Java consumer code can call this method to register itself as a listener for serial data
250     * events.
251     * </p>
252     * 
253     * @see #SerialListener
254     * @see #SerialDataEvent
255     * 
256     * @param listener <p>
257     *            A class instance that implements the SerialListener interface.
258     *            </p>
259     */
260    void addListener(SerialDataListener listener);
261
262    /**
263     * <h1>Remove Serial Event Listener</h1>
264     * 
265     * <p>
266     * Java consumer code can call this method to unregister itself as a listener for serial data
267     * events.
268     * </p>
269     * 
270     * @see #SerialListener
271     * @see #SerialDataEvent
272     * 
273     * @param listener <p>
274     *            A class instance that implements the SerialListener interface.
275     *            </p>
276     */
277    void removeListener(SerialDataListener listener);
278}
