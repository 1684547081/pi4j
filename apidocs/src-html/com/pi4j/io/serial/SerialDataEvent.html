017package com.pi4j.io.serial;
018
019import java.util.EventObject;
020
021/**
022 * <h1>Serial Data Event</h1>
023 * 
024 * <p>
025 * This class provides the serial data event object.
026 * </p>
027 * 
028 * <p>
029 * Before using the Pi4J library, you need to ensure that the Java VM in configured with access to
030 * the following system libraries:
031 * <ul>
032 * <li>pi4j</li>
033 * <li>wiringPi</li>
034 * </ul>
035 * <blockquote> This library depends on the wiringPi native system library.</br> (developed by
036 * Gordon Henderson @ <a href="https://projects.drogon.net/">https://projects.drogon.net/</a>)
037 * </blockquote>
038 * </p>
039 * 
040 * @see #Serial
041 * @see #SerialDataListener
042 * @see #SerialFactory
043 * 
044 * @see <a href="http://www.pi4j.com/">http://www.pi4j.com/</a>
045 * @author Robert Savage (<a
046 *         href="http://www.savagehomeautomation.com">http://www.savagehomeautomation.com</a>)
047 */
048public class SerialDataEvent extends EventObject
049{
050    private static final long serialVersionUID = 1L;
051    private String data;
052
053    /**
054     * <h1>Default event constructor</h1>
055     * 
056     * @param obj <p>
057     *            The com.pi4j.io.serial.Serial class instance that initiates this event.
058     *            </p>
059     * @param data <p>
060     *            The data received.
061     *            </p>
062     */
063    public SerialDataEvent(Object obj, String data)
064    {
065        super(obj);
066        this.data = data;
067    }
068
069    /**
070     * <p>
071     * Get the data string received.
072     * </p>
073     * 
074     * @return <p>
075     *         The data string received.
076     *         </p>
077     */
078    public String getData()
079    {
080        return data;
081    }
082}
