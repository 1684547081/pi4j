031import java.io.IOException;
032import java.nio.ByteBuffer;
033import java.nio.CharBuffer;
034import java.nio.charset.Charset;
035import java.nio.charset.StandardCharsets;
036import java.util.EventObject;
037
038/**
039 * <p> This class provides the serial data event object.</p>
040 * 
041 * <p>
042 * Before using the Pi4J library, you need to ensure that the Java VM in configured with access to
043 * the following system libraries:
044 * <ul>
045 * <li>pi4j</li>
046 * <li>wiringPi</li>
047 * </ul>
048 * <blockquote> This library depends on the wiringPi native system library.</br> (developed by
049 * Gordon Henderson @ <a href="http://wiringpi.com/">http://wiringpi.com/</a>)
050 * </blockquote>
051 * </p>
052 * 
053 * @see com.pi4j.io.serial.Serial
054 * @see SerialDataEventListener
055 * @see com.pi4j.io.serial.SerialDataReader
056 * @see com.pi4j.io.serial.SerialFactory
057 * 
058 * @see <a href="http://www.pi4j.com/">http://www.pi4j.com/</a>
059 * @author Robert Savage (<a
060 *         href="http://www.savagehomeautomation.com">http://www.savagehomeautomation.com</a>)
061 */
062public class SerialDataEvent extends EventObject {
063
064    private static final long serialVersionUID = 1L;
065    private final Serial serial;
066    private byte[] cachedData = null;
067
068    /**
069     * Default event constructor.
070     */
071    public SerialDataEvent(Serial serial) {
072        super(serial);
073        this.serial = serial;
074    }
075
076    /**
077     * Default event constructor.
078     */
079    public SerialDataEvent(Serial serial, byte[] data) {
080        this(serial);
081        this.cachedData = data;
082    }
083
084    /**
085     * Get the serial interface instance
086     *
087     * @return serial interface
088     */
089    public Serial getSerial(){
090        return this.serial;
091    }
092
093    /**
094     * Get an instance of the serial data reader
095     *
096     * @return serial data reader interface
097     */
098    public SerialDataReader getReader(){
099        return this.serial;
100    }
101
102    /**
103     * Get the number of bytes available in the serial data
104     *
105     * @return number of bytes available
106     * @throws IOException
107     */
108    public int length() throws IOException {
109        if(cachedData != null){
110            return cachedData.length;
111        }
112        return getReader().available();
113    }
114
115    /**
116     * Get all the bytes (byte-array) available in the serial data receive buffer
117     *
118     * @return byte array containing all bytes available in serial data receive buffer
119     * @throws IOException
120     */
121    public byte[] getBytes() throws IOException {
122        if(cachedData == null){
123            if(getReader().available() > 0) {
124                cachedData = getReader().read();
125            }
126            else{
127                cachedData = new byte[0];
128            }
129        }
130        return cachedData;
131    }
132
133    /**
134     * Get all the bytes (byte-buffer) available in the serial data receive buffer
135     *
136     * @return ByteBuffer containing all bytes available in serial data receive buffer
137     * @throws IOException
138     */
139    public ByteBuffer getByteBuffer() throws IOException {
140        return ByteBuffer.wrap(getBytes());
141    }
142
143    /**
144     * Get a string representation of the bytes available in the serial data receive buffer
145     *
146     * @param charset the character-set used to construct the string from the underlying byte array
147     * @return string of data from serial data receive buffer
148     * @throws IOException
149     */
150    public String getString(Charset charset) throws IOException {
151        return getCharBuffer(charset).toString();
152    }
153
154    /**
155     * Get an ASCII string representation of the bytes available in the serial data receive buffer
156     *
157     * @return ASCII string of data from serial data receive buffer
158     * @throws IOException
159     */
160    public String getAsciiString() throws IOException {
161        return getCharBuffer(StandardCharsets.US_ASCII).toString();
162    }
163
164    final protected static char[] hexArray = "0123456789ABCDEF".toCharArray();
165
166    /**
167     * Get a HEX string representation of the bytes available in the serial data receive buffer
168     *
169     * @return HEX string of comma separated data bytes from serial data receive buffer
170     * @throws IOException
171     */
172    public String getHexByteString() throws IOException {
173        return getHexByteString(null, ",", null);
174    }
175
176    /**
177     * Get a HEX string representation of the bytes available in the serial data receive buffer
178     *
179     * @param prefix optional prefix string to append before each data byte
180     * @param separator  optional separator string to append in between each data byte sequence
181     * @param suffix optional suffix string to append after each data byte
182     * @return HEX string of data bytes from serial data receive buffer
183     * @throws IOException
184     */
185    public String getHexByteString(CharSequence prefix, CharSequence separator, CharSequence suffix) throws IOException {
186        byte data[] = getBytes();
187        StringBuilder sb = new StringBuilder();
188        for (int i = 0; i < data.length; i++)
189        {
190            if(i > 0) sb.append(separator);
191            int v = data[i] & 0xff;
192            if(prefix != null) sb.append(prefix);
193            sb.append(hexArray[v >> 4]);
194            sb.append(hexArray[v & 0xf]);
195            if(suffix != null) sb.append(suffix);
196        }
197        return sb.toString();
198    }
199
200    /**
201     * Get a character buffer of the bytes available in the serial data receive buffer
202     *
203     * @param charset the character-set used to construct the character buffer from the underlying byte array
204     * @return CharBuffer of data from serial data receive buffer
205     * @throws IOException
206     */
207    public CharBuffer getCharBuffer(Charset charset) throws IOException {
208        return charset.decode(getByteBuffer());
209    }
210
211    /**
212     * <p>discard/drain all available bytes from the serial data receive buffer</p>
213     */
214    public void discardData() throws IllegalStateException, IOException{
215        getReader().discardData();
216    }
217
218}
