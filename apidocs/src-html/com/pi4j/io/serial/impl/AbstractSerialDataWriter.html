032import java.io.ByteArrayOutputStream;
033import java.io.IOException;
034import java.io.InputStream;
035import java.nio.ByteBuffer;
036import java.nio.CharBuffer;
037import java.nio.charset.Charset;
038import java.nio.charset.StandardCharsets;
039import java.util.Collection;

041public abstract class AbstractSerialDataWriter implements SerialDataWriter {
042
043    // ----------------------------------------
044    // WRITE OPERATIONS

046
047    /**
048     * <p>Sends an array of bytes to the serial port/device identified by the given file descriptor.</p>
049     *
050     * @param data
051     *            A ByteBuffer of data to be transmitted.
052     * @param offset
053     *            The starting index (inclusive) in the array to send from.
054     * @param length
055     *            The number of bytes from the byte array to transmit to the serial port.
056     */
057    public abstract void write(byte[] data, int offset, int length) throws IllegalStateException, IOException;
058
059    /**
060     * <p>Sends one of more bytes to the serial device identified by the given file descriptor.</p>
061     *
062     * @param data
063     *            One or more bytes (or an array) of data to be transmitted. (variable-length-argument)
064     */
065    public void write(byte ... data) throws IllegalStateException, IOException{
066        write(data, 0, data.length);
067    }
068
069    /**
070     * <p>Sends one of more bytes arrays to the serial device identified by the given file descriptor.</p>
071     *
072     * @param data
073     *            One or more byte arrays of data to be transmitted. (variable-length-argument)
074     */
075    public void write(byte[] ... data) throws IllegalStateException, IOException{
076        for(byte[] single : data) {
077            write(single);
078        }
079    }
080
081    /**
082     * Read the content of byte buffer and write the data to the serial port transmit buffer.
083     * (The buffer is read from the current position up to the 'limit' value, not the 'capacity'.  You may need to
084     * rewind() or flip() the byte buffer if you have just written to it.)
085     *
086     * @param data
087     *            A ByteBuffer of data to be transmitted.
088     */
089    public void write(ByteBuffer... data) throws IllegalStateException, IOException{
090        // write each byte buffer to the serial port
091        for(ByteBuffer single : data) {
092
093            // read the byte buffer from the current position up to the limit
094            byte[] payload = new byte[single.remaining()];
095            single.get(payload);
096
097            write(payload);
098        }
099    }
100
101    /**
102     * Read content from an input stream of data and write it to the serial port transmit buffer.
103     *
104     * @param input
105     *          An InputStream of data to be transmitted
106     */
107    public void write(InputStream input) throws IllegalStateException, IOException{
108        // ensure bytes are available
109        if(input.available() <= 0){
110            throw new IOException("No available bytes in input stream to write to serial port.");
111        }
112
113        ByteArrayOutputStream buffer = new ByteArrayOutputStream();
114        int length;
115        byte[] data = new byte[1024];
116        while ((length = input.read(data, 0, data.length)) != -1) {
117            buffer.write(data, 0, length);
118        }
119        buffer.flush();
120
121        write(buffer.toByteArray());
122    }

124    /**
125     * <p>Sends an array of characters to the serial port/device identified by the given file descriptor.</p>
126     *
127     * @param charset
128     *           The character set to use for encoding/decoding bytes to/from text characters
129     * @param data
130     *           An array of chars to be decoded into bytes and transmitted.
131     * @param offset
132     *           The starting index (inclusive) in the array to send from.
133     * @param length
134     *           The number of characters from the char array to transmit to the serial port.
135     */
136    public void write(Charset charset, char[] data, int offset, int length) throws IllegalStateException, IOException{
137        write(charset, CharBuffer.wrap(data, offset, length));
138    }
139
140    /**
141     * <p>Sends an array of characters to the serial port/device identified by the given file descriptor.</p>
142     *
143     * @param charset
144     *           The character set to use for encoding/decoding bytes to/from text characters
145     * @param data
146     *           One or more characters (or an array) of data to be transmitted. (variable-length-argument)
147     */
148    public void write(Charset charset, char ... data) throws IllegalStateException, IOException{
149        write(charset, CharBuffer.wrap(data));
150    }
151
152    /**
153     * <p>Sends an array of ASCII characters to the serial port/device identified by the given file descriptor.</p>
154     *
155     * @param data
156     *           One or more ASCII characters (or an array) of data to be transmitted. (variable-length-argument)
157     */
158    public void write(char ... data) throws IllegalStateException, IOException{
159        write(StandardCharsets.US_ASCII, CharBuffer.wrap(data));
160    }
161
162    /**
163     * <p>Sends one or more CharBuffers to the serial port/device identified by the given file descriptor.</p>
164     *
165     * @param charset
166     *           The character set to use for encoding/decoding bytes to/from text characters
167     * @param data
168     *           One or more CharBuffers (or an array) of data to be transmitted. (variable-length-argument)
169     */
170    public void write(Charset charset, CharBuffer... data) throws IllegalStateException, IOException{
171        for(CharBuffer single : data) {
172            write(charset.encode(single));
173        }
174    }
175
176    /**
177     * <p>Sends one or more ASCII CharBuffers to the serial port/device identified by the given file descriptor.</p>
178     *
179     * @param data
180     *           One or more ASCII CharBuffers (or an array) of data to be transmitted. (variable-length-argument)
181     */
182    public void write(CharBuffer ... data) throws IllegalStateException, IOException{
183        write(StandardCharsets.US_ASCII, data);
184    }
185
186    /**
187     * <p>Sends one or more string objects to the serial port/device identified by the given file descriptor.</p>
188     *
189     * @param charset
190     *           The character set to use for encoding/decoding bytes to/from text characters
191     * @param data
192     *           One or more string objects (or an array) of data to be transmitted. (variable-length-argument)
193     */
194    public void write(Charset charset, CharSequence ... data) throws IllegalStateException, IOException{
195        for(CharSequence single : data) {
196            write(charset.encode(CharBuffer.wrap(single)));
197        }
198    }
199
200    /**
201     * <p>Sends one or more ASCII string objects to the serial port/device identified by the given file descriptor.</p>
202     *
203     * @param data
204     *           One or more ASCII string objects (or an array) of data to be transmitted. (variable-length-argument)
205     */
206    public void write(CharSequence ... data) throws IllegalStateException, IOException{
207        write(StandardCharsets.US_ASCII, data);
208    }
209
210    /**
211     * <p>Sends a collection of string objects to the serial port/device identified by the given file descriptor.</p>
212     *
213     * @param charset
214     *           The character set to use for encoding/decoding bytes to/from text characters
215     * @param data
216     *           A collection of string objects (or an array) of data to be transmitted. (variable-length-argument)
217     */
218    public void write(Charset charset, Collection<? extends CharSequence> data) throws IllegalStateException, IOException{
219        for(CharSequence single : data) {
220            write(charset.encode(CharBuffer.wrap(single)));
221        }
222    }
223
224    /**
225     * <p>Sends a collection of ASCII string objects to the serial port/device identified by the given file descriptor.</p>
226     *
227     * @param data
228     *           A collection of string objects (or an array) of data to be transmitted. (variable-length-argument)
229     */
230    public void write(Collection<? extends CharSequence> data) throws IllegalStateException, IOException{
231        write(StandardCharsets.US_ASCII, data);
232    }

234    protected CharSequence appendNewLine(CharSequence data){
235        String separator = System.getProperty("line.separator");
236        StringBuilder builder = new StringBuilder(data);
237        builder.append(separator);
238        return builder.toString();
239    }
240
241    /**
242     * <p>Sends one or more string objects each appended with a line terminator (CR+LF) to the serial port/device.</p>
243     *
244     * @param charset
245     *           The character set to use for encoding/decoding bytes to/from text characters
246     * @param data
247     *           One or more string objects (or an array) of data to be transmitted. (variable-length-argument)
248     */
249    public void writeln(Charset charset, CharSequence ... data) throws IllegalStateException, IOException{
250        for(CharSequence single : data) {
251            write(charset.encode(CharBuffer.wrap(appendNewLine(single))));
252        }
253    }
254
255    /**
256     * <p>Sends one or more ASCII string objects each appended with a line terminator (CR+LF) to the serial port/device.</p>
257     *
258     * @param data
259     *           One or more ASCII string objects (or an array) of data to be transmitted. (variable-length-argument)
260     */
261    public void writeln(CharSequence ... data) throws IllegalStateException, IOException{
262        writeln(StandardCharsets.US_ASCII, data);
263    }
264
265    /**
266     * <p>Sends a collection of string objects each appended with a line terminator (CR+LF) to the serial port/device.</p>
267     *
268     * @param charset
269     *           The character set to use for encoding/decoding bytes to/from text characters
270     * @param data
271     *           A collection of string objects (or an array) of data to be transmitted. (variable-length-argument)
272     */
273    public void writeln(Charset charset, Collection<? extends CharSequence> data) throws IllegalStateException, IOException{
274        for(CharSequence single : data) {
275            write(charset.encode(CharBuffer.wrap(appendNewLine(single))));
276        }
277    }
278
279    /**
280     * <p>Sends a collection of ASCII string objects each appended with a line terminator (CR+LF) to the serial port/device.</p>
281     *
282     * @param data
283     *           A collection of ASCII string objects (or an array) of data to be transmitted. (variable-length-argument)
284     */
285    public void writeln(Collection<? extends CharSequence> data) throws IllegalStateException, IOException{
286        writeln(StandardCharsets.US_ASCII, data);
