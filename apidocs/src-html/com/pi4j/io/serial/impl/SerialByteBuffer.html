030import java.io.IOException;
031import java.io.InputStream;
032import java.nio.BufferOverflowException;
033
034/**
035 * This class implements a dynamic expanding byte buffer to
036 * accommodate new data received from the serial port
037 *
038 * Adapted from sources at:
039 * http://ostermiller.org/utils/src/CircularByteBuffer.java.html
040 * Stephen Ostermiller http://ostermiller.org/contact.pl?regarding=Java+Utilities
041 *
042 */
043public class SerialByteBuffer {
044
045    public static int DEFAULT_BUFFER_SCALE_FACTOR = 2;
046    public static int DEFAULT_INITIAL_BUFFER_SIZE = 4096;
047
048    private InputStream stream = new SerialByteBufferInputStream();
049    private volatile int writeIndex = 0;
050    private volatile int readIndex = 0;
051    private byte[] buffer;
052
053    public SerialByteBuffer(){
054        // initialize buffer with default capacity
055        this(DEFAULT_INITIAL_BUFFER_SIZE);
056    }
057
058    public SerialByteBuffer(int initialCapacity){
059        // initialize buffer with user provided capacity
060        buffer = new byte[initialCapacity];
061    }
062
063    public synchronized void clear(){
064        // reset read and write index pointers
065        readIndex = writeIndex = 0;
066    }
067
068    public InputStream getInputStream(){
069        // return the input stream
070        return stream;
071    }
072
073    public synchronized int capacity(){
074        // return the buffer's total capacity
075        return buffer.length;
076    }
077
078    public synchronized int remaining(){
079        // return the number of (unused) bytes still available in the current buffer's capacity
080        if (writeIndex < readIndex)
081            return (readIndex - writeIndex - 1);
082        return ((buffer.length - 1) - (writeIndex - readIndex));
083    }
084
085    public synchronized int available(){
086        // return the number of bytes that are ready to be read
087        if (readIndex <= writeIndex)
088            return (writeIndex - readIndex);
089        return (buffer.length - (readIndex - writeIndex));
090    }
091
092    private void resize(int length) {
093
094        int min_capacity = buffer.length + length;
095        int new_capacity = buffer.length;
096
097        // double the capacity until the buffer is large enough to accommodate the new demand
098        while (new_capacity < min_capacity) {
099            new_capacity *= DEFAULT_BUFFER_SCALE_FACTOR;
100        }
101
102        // create a new buffer that can hold the newly determined
103        // capacity and copy the bytes from the old buffer into the new buffer
104        byte[] new_buffer = new byte[new_capacity];
105        System.arraycopy(buffer, readIndex, new_buffer, 0, writeIndex);
106
107        // update pointers
108        buffer = new_buffer; // old buffer should get garbage collected
109        readIndex = 0;
110        writeIndex = available();
111    }
112
113    public  void write(byte[] data) throws IOException, BufferOverflowException {
114        write(data, 0, data.length);
115    }
116
117    public  void write(byte[] data, int offset, int length) throws IOException {
118        while (length > 0) {
119            int remaining_space = remaining();
120            if(remaining_space < length) {
121                resize(length);
122            }
123            int realLen = Math.min(length, remaining_space);
124            int firstLen = Math.min(realLen, buffer.length - writeIndex);
125            int secondLen = Math.min(realLen - firstLen, buffer.length - readIndex - 1);
126            int written = firstLen + secondLen;
127            if (firstLen > 0) {
128                System.arraycopy(data, offset, buffer, writeIndex, firstLen);
129            }
130            if (secondLen > 0) {
131                System.arraycopy(data, offset + firstLen, buffer, 0, secondLen);
132                writeIndex = secondLen;
133            } else {
134                writeIndex += written;
135            }
136            if (writeIndex == buffer.length) {
137                writeIndex = 0;
138            }
139            offset += written;
140            length -= written;
141        }
142        if (length > 0) {
143            try {
144                Thread.sleep(100);
145            } catch (Exception x) {
146                throw new IOException("Waiting for available space in buffer interrupted.");
147            }
148        }
149    }
150
151    protected class SerialByteBufferInputStream extends InputStream {
152
153        @Override
154        public int available() throws IOException {
155            synchronized (SerialByteBuffer.this){
156                return (SerialByteBuffer.this.available());
157            }
158        }
159
160        @Override
161        public int read() throws IOException {
162            while (true){
163                synchronized (SerialByteBuffer.this){
164                    int available = SerialByteBuffer.this.available();
165                    if (available > 0){
166                        int result = buffer[readIndex] & 0xff; // we only care about fist 8 bits
167                        readIndex++; // increment read index position
168                        // if the read index reaches the maximum buffer capacity, then rollover to zero index
169                        if (readIndex == buffer.length)
170                            readIndex = 0;
171                        return result;
172                    }
173                }
174                try {
175                    Thread.sleep(100);
176                } catch(Exception x){
177                    throw new IOException("Blocking read operation interrupted.");
178                }
179            }
180        }
181
182        @Override
183        public int read(byte[] data) throws IOException {
184            return read(data, 0, data.length);
185        }
186
187        @Override
188        public int read(byte[] data, int off, int len) throws IOException {
189            while (true){
190                synchronized (SerialByteBuffer.this){
191                    int available = SerialByteBuffer.this.available();
192                    if (available > 0){
193                        int length = Math.min(len, available);
194                        int firstLen = Math.min(length, buffer.length - readIndex);
195                        int secondLen = length - firstLen;
196                        System.arraycopy(buffer, readIndex, data, off, firstLen);
197                        if (secondLen > 0){
198                            System.arraycopy(buffer, 0, data, off+firstLen,  secondLen);
199                            readIndex = secondLen;
200                        } else {
201                            readIndex += length;
202                        }
203                        if (readIndex == buffer.length) {
204                            readIndex = 0;
205                        }
206
207                        return length;
208                    }
209                }
210                try {
211                    Thread.sleep(100);
212                } catch(Exception x){
213                    throw new IOException("Blocking read operation interrupted.");
214                }
215            }
216        }
217
218        @Override
219        public long skip(long n) throws IOException, IllegalArgumentException {
220            while (true){
221                synchronized (SerialByteBuffer.this){
222                    int available = SerialByteBuffer.this.available();
223                    if (available > 0){
224                        int length = Math.min((int)n, available);
225                        int firstLen = Math.min(length, buffer.length - readIndex);
226                        int secondLen = length - firstLen;
227                        if (secondLen > 0){
228                            readIndex = secondLen;
229                        } else {
230                            readIndex += length;
231                        }
232                        if (readIndex == buffer.length) {
233                            readIndex = 0;
234                        }
235                        return length;
236                    }
237                }
238                try {
239                    Thread.sleep(100);
240                } catch(Exception x){
241                    throw new IOException("Blocking read operation interrupted.");
242                }
243            }
244        }
245    }
