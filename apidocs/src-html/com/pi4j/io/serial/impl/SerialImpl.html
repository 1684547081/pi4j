017package com.pi4j.io.serial.impl;
018
019import java.util.Vector;
020
021import com.pi4j.io.serial.Serial;
022import com.pi4j.io.serial.SerialDataListener;
023
024/**
025 * <h1>Serial Implementation Class</h1>
026 * 
027 * <p>
028 * This implementation class implements the 'Serial' interface using the WiringPi Serial library.
029 * </p>
030 * 
031 * <p>
032 * Before using the Pi4J library, you need to ensure that the Java VM in configured with access to
033 * the following system libraries:
034 * <ul>
035 * <li>pi4j</li>
036 * <li>wiringPi</li>
037 * </ul>
038 * <blockquote> This library depends on the wiringPi native system library.</br> (developed by
039 * Gordon Henderson @ <a href="https://projects.drogon.net/">https://projects.drogon.net/</a>)
040 * </blockquote>
041 * </p>
042 * 
043 * @see #Serial
044 * @see #SerialDataEvent
045 * @see #SerialDataListener
046 * @see #SerialFactory
047 * 
048 * @see <a href="http://www.pi4j.com/">http://www.pi4j.com/</a>
049 * @author Robert Savage (<a
050 *         href="http://www.savagehomeautomation.com">http://www.savagehomeautomation.com</a>)
051 */
052public class SerialImpl implements Serial
053{
054    private int fileDescriptor = -1;
055    private Vector<SerialDataListener> listeners = new Vector<SerialDataListener>();
056    private SerialDataMonitorThread monitor;
057
058    public int open(String device, int baudRate)
059    {
060        fileDescriptor = com.pi4j.wiringpi.Serial.serialOpen(device, baudRate);
061        return fileDescriptor;
062    }
063
064    public boolean isOpen()
065    {
066        return (fileDescriptor >= 0);
067    }
068
069    public void close()
070    {
071        com.pi4j.wiringpi.Serial.serialClose(fileDescriptor);
072    }
073
074    public void flush()
075    {
076        com.pi4j.wiringpi.Serial.serialFlush(fileDescriptor);
077    }
078
079    public char read()
080    {
081        return (char) com.pi4j.wiringpi.Serial.serialGetchar(fileDescriptor);
082    }
083
084    public char readLine()
085    {
086        return ' ';
087    }
088
089    public void write(char data)
090    {
091        com.pi4j.wiringpi.Serial.serialPutchar(fileDescriptor, data);
092    }
093
094    public void write(char data[])
095    {
096        write(new String(data));
097    }
098
099    public void write(byte data)
100    {
101        com.pi4j.wiringpi.Serial.serialPutchar(fileDescriptor, (char) data);
102    }
103
104    public void write(byte data[])
105    {
106        write(new String(data));
107    }
108
109    public void write(String data)
110    {
111        // break data into packets of 1024 bytes
112        int position = 0;
113        while (position < data.length())
114        {
115            int length = 1024;
116            if (position + 1024 > data.length())
117                com.pi4j.wiringpi.Serial.serialPuts(fileDescriptor, data.substring(position));
118            else
119                com.pi4j.wiringpi.Serial.serialPuts(fileDescriptor,
120                                                    data.substring(position, (position + length)));
121            position += length;
122        }
123    }
124
125    public void write(String data, String... args)
126    {
127        write(String.format(data, (Object[]) args));
128    }
129
130    public int availableBytes()
131    {
132        return com.pi4j.wiringpi.Serial.serialDataAvail(fileDescriptor);
133    }
134
135    /**
136     * <h1>Add Serial Event Listener</h1>
137     * 
138     * <p>
139     * Java consumer code can call this method to register itself as a listener for serial data
140     * events.
141     * </p>
142     * 
143     * @see #SerialListener
144     * @see #SerialDataEvent
145     * 
146     * @param listener <p>
147     *            A class instance that implements the SerialListener interface.
148     *            </p>
149     */
150    public synchronized void addListener(SerialDataListener listener)
151    {
152        // add the new listener to the list of listeners
153        listeners.addElement(listener);
154
155        // if there is not a current listening monitor thread running,
156        // then lets start it now
157        if (monitor == null || monitor.isAlive() == false)
158        {            
159            monitor = new SerialDataMonitorThread(this, listeners);
160            monitor.start();
161        }
162    }
163
164    /**
165     * <h1>Remove Serial Event Listener</h1>
166     * 
167     * <p>
168     * Java consumer code can call this method to unregister itself as a listener for serial data
169     * events.
170     * </p>
171     * 
172     * @see #SerialListener
173     * @see #SerialDataEvent
174     * 
175     * @param listener <p>
176     *            A class instance that implements the SerialListener interface.
177     *            </p>
178     */
179    public synchronized void removeListener(SerialDataListener listener)
180    {
181        // remove the listener from the list of listeners
182        listeners.removeElement(listener);
183        
184        // if there are not more listeners, then exit and destroy 
185        // the monitor thread now
186        if(listeners.isEmpty() && monitor != null)
187        {
188            monitor.exit();
189            monitor = null;
190        }
191    }
192}
