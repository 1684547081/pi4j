029import java.io.File;
030import java.io.FileNotFoundException;
031import java.io.IOException;
032import java.io.InputStream;
033import java.nio.file.Files;
034import java.nio.file.Path;
035import java.nio.file.Paths;
036import java.nio.file.StandardCopyOption;
037import java.util.Set;
038import java.util.TreeSet;
039import java.util.logging.ConsoleHandler;
040import java.util.logging.FileHandler;
041import java.util.logging.Level;
042import java.util.logging.Logger;
043
044public class NativeLibraryLoader {
045
046        private static final Set<String> loadedLibraries = new TreeSet<String>();
047        private static final Logger logger = Logger.getLogger(NativeLibraryLoader.class.getName());
048        private static boolean initialized;
049
050        // private constructor
051        private NativeLibraryLoader() {
052                // forbid object construction
053        }
054
055        public static synchronized void load(String fileName) {
056                // check for debug property; if found enable all logging levels
057                if (!initialized) {
058                        initialized = true;
059                        if (System.getProperty("pi4j.debug") != null) {
060                                logger.setLevel(Level.ALL);
061                                try {
062                                        // create an appending file handler
063                                        FileHandler fileHandler = new FileHandler("pi4j.log");
064                                        fileHandler.setLevel(Level.ALL);
065                                        ConsoleHandler consoleHandler = new ConsoleHandler();
066                                        consoleHandler.setLevel(Level.ALL);
067
068                                        // add to the desired loggers
069                                        logger.addHandler(fileHandler);
070                                        logger.addHandler(consoleHandler);
071                                } catch (IOException e) {
072                                        System.err.println("Unable to setup logging to debug. No logging will be done. Error: ");
073                                        e.printStackTrace();
074                                }
075                        }
076                }
077
078                // first, make sure that this library has not already been previously loaded
079                if (loadedLibraries.contains(fileName)) {
080                        logger.fine("Library [" + fileName + "] has already been loaded; no need to load again.");
081                        return;
082                }
083
084                loadedLibraries.add(fileName);
085
086                String path = "/lib/" + fileName;
087                logger.fine("Attempting to load [" + fileName + "] using path: [" + path + "]");
088                try {
089                        loadLibraryFromClasspath(path);
090                        logger.fine("Library [" + fileName + "] loaded successfully using embedded resource file: [" + path + "]");
091                } catch (Exception | UnsatisfiedLinkError e) {
092                        logger.log(Level.SEVERE, "Unable to load [" + fileName + "] using path: [" + path + "]", e);
093                        // either way, we did what we could, no need to remove now the library from the loaded libraries since we run inside one VM and nothing could possibly change, so there is no point in
094                        // trying out this logic again
095                }
096        }
097
098        /**
099         * Loads library from classpath
100         * 
101         * The file from classpath is copied into system temporary directory and then loaded. The temporary file is deleted after exiting. Method uses String as filename because the pathname is
102         * "abstract", not system-dependent.
103         * 
104         * @param filename
105         *            The filename in classpath as an absolute path, e.g. /package/File.ext (could be inside jar)
106         * @throws IOException
107         *             If temporary file creation or read/write operation fails
108         * @throws IllegalArgumentException
109         *             If source file (param path) does not exist
110         * @throws IllegalArgumentException
111         *             If the path is not absolute or if the filename is shorter than three characters (restriction of {@see File#createTempFile(java.lang.String, java.lang.String)}).
112         */
113        public static void loadLibraryFromClasspath(String path) throws IOException {
114                Path inputPath = Paths.get(path);
115
116                if (!inputPath.isAbsolute()) {
117                        throw new IllegalArgumentException("The path has to be absolute, but found: " + inputPath);
118                }
119
120                String fileNameFull = inputPath.getFileName().toString();
121                int dotIndex = fileNameFull.indexOf('.');
122                if (dotIndex < 0 || dotIndex >= fileNameFull.length() - 1) {
123                        throw new IllegalArgumentException("The path has to end with a file name and extension, but found: " + fileNameFull);
124                }
125
126                String fileName = fileNameFull.substring(0, dotIndex);
127                String extension = fileNameFull.substring(dotIndex);
128
129                Path target = Files.createTempFile(fileName, extension);
130                File targetFile = target.toFile();
131                targetFile.deleteOnExit();
132
133                try (InputStream source = NativeLibraryLoader.class.getResourceAsStream(inputPath.toString())) {
134                        if (source == null) {
135                                throw new FileNotFoundException("File " + inputPath + " was not found in classpath.");
136                        }
137                        Files.copy(source, target, StandardCopyOption.REPLACE_EXISTING);
138                }
139                // Finally, load the library
140                System.load(target.toAbsolutePath().toString());
141        }
142}
