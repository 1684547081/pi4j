029import java.io.*;
030import java.net.URL;
031import java.util.ArrayList;
032import java.util.Collections;
033import java.util.List;
034import java.util.logging.ConsoleHandler;
035import java.util.logging.FileHandler;
036import java.util.logging.Level;
037import java.util.logging.Logger;
038
039public class NativeLibraryLoader {
040
041    private static List<String> loadedLibraries = null;
042    private static Logger logger = Logger.getLogger("com.pi4j.util.NativeLibraryLoader");
043    private static boolean initialized  = false;
044
045    // private constructor 
046    private NativeLibraryLoader() {
047        // forbid object construction 
048    }
049    
050    public static synchronized void load(String libraryName) {
051        load(libraryName, null);
052    }
053
054    public static synchronized void load(String libraryName, String fileName) {
055        ConsoleHandler consoleHandler;
056        FileHandler fileHandler;
057
058        // check for debug property; if found enabled all logging levels
059        if (!initialized) {
060            initialized = true;
061            String debug = System.getProperty("pi4j.debug");
062            if (debug != null) {
063                logger.setLevel(Level.ALL);
064                try {
065                    // create an appending file handler
066                    fileHandler = new FileHandler("pi4j.log");
067                    fileHandler.setLevel(Level.ALL);
068                    consoleHandler = new ConsoleHandler();
069                    consoleHandler.setLevel(Level.ALL);
070    
071                    // add to the desired loggers
072                    logger.addHandler(fileHandler);
073                    logger.addHandler(consoleHandler);
074                } 
075                catch (IOException e) {
076                    // do nothing
077                }
078            }
079        }
080        
081        // debug
082        if (fileName == null || fileName.length() == 0) {
083            logger.fine("Load library [" + libraryName + "] (no alternate embedded file provided)");
084        } else {
085            logger.fine("Load library [" + libraryName + "] (alternate embedded file: " + fileName + ")");
086        }
087        // create instance if null
088        if (loadedLibraries == null) {
089            loadedLibraries = Collections.synchronizedList(new ArrayList<String>());
090        }
091        // first, make sure that this library has not already been previously loaded
092        if (loadedLibraries.contains(libraryName)) {
093            // debug
094            logger.fine("Library [" + libraryName + "] has already been loaded; no need to load again.");            
095        } else {
096            // ---------------------------------------------
097            // ATTEMPT LOAD FROM SYSTEM LIBS
098            // ---------------------------------------------
099            
100            // assume library loaded successfully, add to tracking collection
101            loadedLibraries.add(libraryName);
102            
103            try {
104                // debug
105                logger.fine("Attempting to load library [" + libraryName + "] using the System.loadLibrary(name) method.");            
106                
107                // attempt to load the native library from the system classpath loader
108                System.loadLibrary(libraryName);
109                
110                // debug
111                logger.fine("Library [" + libraryName + "] loaded successfully using the System.loadLibrary(name) method.");                            
112            } catch (UnsatisfiedLinkError e) {
113                // if a filename was not provided, then throw exception
114                if (fileName == null) {
115                    // debug
116                    logger.severe("Library [" + libraryName + "] could not be located using the System.loadLibrary(name) method and no embedded file path was provided as an auxillary lookup.");                            
117
118                    // library load failed, remove from tracking collection
119                    loadedLibraries.remove(libraryName);
120                    throw e;
121                }
122
123                // debug
124                logger.fine("Library [" + libraryName + "] could not be located using the System.loadLibrary(name) method; attempting to resolve the library using embedded resources in the JAR file.");                            
125
126                
127                // ---------------------------------------------
128                // ATTEMPT LOAD LIBRARY PACKAGED INSIDE JAR
129                // ---------------------------------------------
130
131                // attempt to get the native library from the JAR file in the 'lib' directory
132                URL resourceUrl = NativeLibraryLoader.class.getResource("/lib/" + fileName);
133
134                try {
135                    // load library file from embedded resource
136                    loadLibraryFromResource(resourceUrl, libraryName, fileName);
137                    
138                    // debug
139                    logger.fine("Library [" + libraryName + "] loaded successfully using embedded resource file: [" + resourceUrl.toString() + "]");
140                } 
141                
142                catch(Exception|UnsatisfiedLinkError ex) {
143                    // debug
144                    logger.severe("Failed to load library [" + libraryName + "] using the System.load(file) method using embedded resource file: [" + resourceUrl.toString() + "]");
145                    logger.throwing(logger.getName(), "load", ex);
146
147                    // library load failed, remove from tracking collection
148                    loadedLibraries.remove(libraryName);
149
150                    logger.severe("ERROR:  The native library ["
151                                + libraryName
152                                + " : "
153                                + fileName
154                                + "] could not be found in the JVM library path nor could it be loaded from the embedded JAR resource file; you may need to explicitly define the library path '-Djava.library.path' where this native library can be found.");
155                }
156            }
157        }
158    }
159
160    private static void loadLibraryFromResource(URL resourceUrl, String libraryName, String fileName) throws UnsatisfiedLinkError, Exception {
161        // create a 1Kb read buffer
162        byte[] buffer = new byte[1024];
163        int byteCount;
164        
165        // debug
166        logger.fine("Attempting to load library [" + libraryName + "] using the System.load(file) method using embedded resource file: [" + resourceUrl.toString() + "]");            
167        
168        // open the resource file stream
169        InputStream inputStream = resourceUrl.openStream();
170
171        // get the system temporary directory path
172        File tempDirectory = new File(System.getProperty("java.io.tmpdir"));
173        
174        // check to see if the temporary path exists
175        if (!tempDirectory.exists()) {
176            // debug
177            logger.warning("The Java system temporary path [" + tempDirectory.getAbsolutePath() + "] does not exist.");            
178            
179            // instead of the system defined temporary path, let just use the application path
180            tempDirectory = new File("");
181        }
182        
183        // create a temporary file to copy the native library content to
184        File tempFile = new File(tempDirectory.getAbsolutePath() + "/" + fileName);
185        
186        // make sure that this temporary file does not exist; if it does then delete it
187        if (tempFile.exists()) {
188            // debug
189            logger.warning("The temporary file already exists [" + tempFile.getAbsolutePath() + "]; attempting to delete it now.");            
190            
191            // delete file immediately 
192            if(!tempFile.delete()){
193                logger.warning("The temporary file was not deleted [" + tempFile.getAbsolutePath() + "]");
194            }
195        }
196        
197        // create output stream object
198        OutputStream outputStream;
199        
200        try {
201            // create the new file
202            outputStream = new FileOutputStream(tempFile);
203        } catch(FileNotFoundException fnfe) {
204            // error
205            logger.severe("The temporary file [" + tempFile.getAbsolutePath() + "] cannot be created; it is a directory, not a file.");            
206            throw(fnfe);
207        } catch(SecurityException se) {
208            // error
209            logger.severe("The temporary file [" + tempFile.getAbsolutePath() + "] cannot be created; a security exception was detected. " + se.getMessage());            
210            throw(se);
211        }
212        
213        try {
214            // copy the library file content
215            while ((byteCount = inputStream.read(buffer)) >= 0) {
216                outputStream.write(buffer, 0, byteCount);
217            }
218            
219            // flush all write data from stream 
220            outputStream.flush();
221            
222            // close the output stream
223            outputStream.close();                            
224        } catch(IOException ioe) {
225            // error
226            logger.severe("The temporary file [" + tempFile.getAbsolutePath() + "] could not be written to; an IO exception was detected. " + ioe.getMessage());            
227            throw(ioe);                            
228        }
229
230        // close the input stream
231        inputStream.close();
232        
233        try {
234            // attempt to load the new temporary library file
235            System.load(tempFile.getAbsolutePath());
236            
237            try {
238                // ensure that this temporary file is removed when the program exits
239                tempFile.deleteOnExit();
240            } catch(SecurityException dse) {
241                // warning
242                logger.warning("The temporary file [" + tempFile.getAbsolutePath() + "] cannot be flagged for removal on program termination; a security exception was detected. " + dse.getMessage());            
243            }
244        } catch(UnsatisfiedLinkError|Exception ex) {
245            // if unable to load the library and the temporary file
246            // exists; then delete the temporary file immediately
247            if (tempFile.exists()) {
248                // delete file immediately
249                if(!tempFile.delete()){
250                    logger.warning("The temporary file was not deleted [" + tempFile.getAbsolutePath() + "]");
251                }
252            }
253            
254            throw(ex);
255        }
256    }
257}
