017package com.pi4j.wiringpi;
018
019import java.util.EventObject;
020
021/**
022 * <h1>Gpio Interrupt Event</h1>
023 * 
024 * <p>
025 * This class provides the event object for GPIO interrupt state changes.
026 * </p>
027 * 
028 * <p>
029 * Before using the Pi4J library, you need to ensure that the Java VM in configured with access to
030 * the following system libraries:
031 * <ul>
032 * <li>pi4j</li>
033 * <li>wiringPi</li>
034 * </ul>
035 * <blockquote> This library depends on the wiringPi native system library.</br> (developed by
036 * Gordon Henderson @ <a href="https://projects.drogon.net/">https://projects.drogon.net/</a>)
037 * </blockquote>
038 * </p>
039 * 
040 * @see <a href="http://www.pi4j.com/">http://www.pi4j.com/</a>
041 * @author Robert Savage (<a
042 *         href="http://www.savagehomeautomation.com">http://www.savagehomeautomation.com</a>)
043 */
044public class GpioInterruptEvent extends EventObject
045{
046    private static final long serialVersionUID = 1L;
047    private int pin;
048    private boolean state;
049
050    /**
051     * <h1>Default event constructor</h1>
052     * 
053     * @param obj <p>
054     *            Ignore this parameter
055     *            </p>
056     * @param pin <p>
057     *            GPIO pin number (not header pin number; not wiringPi pin number)
058     *            </p>
059     * @param state <p>
060     *            New GPIO pin state.
061     *            </p>
062     */
063    public GpioInterruptEvent(Object obj, int pin, boolean state)
064    {
065        super(obj);
066        this.pin = pin;
067        this.state = state;
068    }
069
070    /**
071     * <p>
072     * Get the pin number that changed and raised this event.
073     * </p>
074     * 
075     * @return <p>
076     *         GPIO pin number (not header pin number; not wiringPi pin number)
077     *         </p>
078     */
079    public int getPin()
080    {
081        return pin;
082    }
083
084    /**
085     * <p>
086     * Get the new pin state raised in this event.
087     * </p>
088     * 
089     * @return <p>
090     *         GPIO pin state (HIGH=true, LOW=false)
091     *         </p>
092     */
093    public boolean getState()
094    {
095        return state;
096    }
097
098    /**
099     * <p>
100     * Get the new pin state value raised in this event.
101     * </p>
102     * 
103     * @return <p>
104     *         GPIO pin state (HIGH=1, LOW=0)
105     *         </p>
106     */
107    public int getStateValue()
108    {
109        return (state == true) ? 1 : 0;
110    }
111}
