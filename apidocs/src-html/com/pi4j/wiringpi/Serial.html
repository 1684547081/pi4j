017package com.pi4j.wiringpi;
018
019/**
020 * <h1>WiringPi Serial Port Communication</h1>
021 * 
022 * <p>
023 * WiringPi includes a simplified serial port handling library. It can use the on-board serial port,
024 * or any USB serial device with no special distinctions between them. You just specify the device
025 * name in the initial open function.
026 * </p>
027 * 
028 * <p>
029 * Note: The file descriptor (fd) returned is a standard Linux filehandle. You can use the standard
030 * read(), write(), etc. system calls on this filehandle as required. E.g. you may wish to write a
031 * larger block of binary data where the serialPutchar() or serialPuts() function may not be the
032 * most appropriate function to use, in which case, you can use write() to send the data.
033 * </p>
034 * 
035 * <p>
036 * Before using the Pi4J library, you need to ensure that the Java VM in configured with access to
037 * the following system libraries:
038 * <ul>
039 * <li>pi4j</li>
040 * <li>wiringPi</li>
041 * </ul>
042 * <blockquote> This library depends on the wiringPi native system library.</br> (developed by
043 * Gordon Henderson @ <a href="https://projects.drogon.net/">https://projects.drogon.net/</a>)
044 * </blockquote>
045 * </p>
046 * 
047 * @see <a href="http://www.pi4j.com/">http://www.pi4j.com/</a>
048 * @see <a
049 *      href="https://projects.drogon.net/raspberry-pi/wiringpi/serial-library/">https://projects.drogon.net/raspberry-pi/wiringpi/serial-library/</a>
050 * @author Robert Savage (<a
051 *         href="http://www.savagehomeautomation.com">http://www.savagehomeautomation.com</a>)
052 */
053public class Serial
054{
055    /**
056     * <p>The default hardware COM port provided via the Raspberry Pi GPIO header.</p>
057     * 
058     * @see #serialOpen(String,int)
059     */
060    public static final String DEFAULT_COM_PORT = "/dev/ttyAMA0";
061
062    // Load the platform library
063    static
064    {
065        System.loadLibrary("pi4j");
066    }
067
068    /**
069     * <h1>int serialOpen (char *device, int baud);</h1>
070     * 
071     * <p>
072     * This opens and initializes the serial device and sets the baud rate. It sets the port into
073     * raw mode (character at a time and no translations), and sets the read timeout to 10 seconds.
074     * The return value is the file descriptor or -1 for any error, in which case errno will be set
075     * as appropriate.
076     * </p>
077     * 
078     * <p>
079     * (ATTENTION: the 'device' argument can only be a maximum of 128 characters.)
080     * </p>
081     * 
082     * @see #DEFAULT_COM_PORT
083     * @see <a
084     *      href="https://projects.drogon.net/raspberry-pi/wiringpi/serial-library/">https://projects.drogon.net/raspberry-pi/wiringpi/serial-library/</a>
085     * 
086     * @param device <p>
087     *            <The device address of the serial port to access. You can use constant
088     *            'DEFAULT_COM_PORT' if you wish to access the default serial port provided via the
089     *            GPIO header./p>
090     * @param baud <p>
091     *            The baud rate to use with the serial port.
092     *            </p>
093     * @return <p>
094     *         The return value is the file descriptor or -1 for any error, in which case errno will
095     *         be set as appropriate.
096     *         </p>
097     */
098    public static native int serialOpen(String device, int baud);
099
100    /**
101     * <h1>void serialClose (int fd);</h1>
102     * 
103     * <p>
104     * Closes the device identified by the file descriptor given.
105     * </p>
106     * 
107     * @see <a
108     *      href="https://projects.drogon.net/raspberry-pi/wiringpi/serial-library/">https://projects.drogon.net/raspberry-pi/wiringpi/serial-library/</a>
109     * @param fd <p>
110     *            The file descriptor of the serial port.
111     *            </p>
112     */
113    public static native void serialClose(int fd);
114
115    /**
116     * <h1>void serialFlush (int fd);</h1>
117     * 
118     * <p>
119     * This discards all data received, or waiting to be send down the given device.
120     * </p>
121     * 
122     * @see <a
123     *      href="https://projects.drogon.net/raspberry-pi/wiringpi/serial-library/">https://projects.drogon.net/raspberry-pi/wiringpi/serial-library/</a>
124     * @param fd <p>
125     *            The file descriptor of the serial port.
126     *            </p>
127     */
128    public static native void serialFlush(int fd);
129
130    /**
131     * <h1>void serialPutchar (int fd, unsigned char c);</h1>
132     * 
133     * <p>
134     * Sends the single byte to the serial device identified by the given file descriptor.
135     * </p>
136     * 
137     * @see <a
138     *      href="https://projects.drogon.net/raspberry-pi/wiringpi/serial-library/">https://projects.drogon.net/raspberry-pi/wiringpi/serial-library/</a>
139     * @param fd <p>
140     *            The file descriptor of the serial port.
141     *            </p>
142     * @param c <p>
143     *            The character to transmit to the serial port.
144     *            </p>
145     */
146    public static native void serialPutchar(int fd, char data);
147
148    /**
149     * <h1>void serialPuts (int fd, char *s);</h1>
150     * 
151     * <p>
152     * Sends the nul-terminated string to the serial device identified by the given file descriptor.
153     * </p>
154     * 
155     * <p>
156     * (ATTENTION: the 'data' argument can only be a maximum of 1024 characters.)
157     * </p>
158     * 
159     * @see <a
160     *      href="https://projects.drogon.net/raspberry-pi/wiringpi/serial-library/">https://projects.drogon.net/raspberry-pi/wiringpi/serial-library/</a>
161     * @param fd <p>
162     *            The file descriptor of the serial port.
163     *            </p>
164     * @param s <p>
165     *            The data string to transmit to the serial port.
166     *            </p>
167     */
168    public static native void serialPuts(int fd, String data);
169
170    /**
171     * <h1>void serialPuts (int fd, String data, String...arguments);</h1>
172     * 
173     * <p>
174     * Sends the nul-terminated formatted string to the serial device identified by the given file
175     * descriptor.
176     * </p>
177     * 
178     * <p>
179     * (ATTENTION: the 'data' argument can only be a maximum of 1024 characters.)
180     * </p>
181     * 
182     * @see <a
183     *      href="https://projects.drogon.net/raspberry-pi/wiringpi/serial-library/">https://projects.drogon.net/raspberry-pi/wiringpi/serial-library/</a>
184     * @param fd <p>
185     *            The file descriptor of the serial port.
186     *            </p>
187     * @param data <p>
188     *            The formatted data string to transmit to the serial port.
189     *            </p>
190     * @param args <p>
191     *            Arguments to the format string.
192     *            </p>
193     */
194    public static void serialPuts(int fd, String data, String... args)
195    {
196        serialPuts(fd, String.format(data, (Object[]) args));
197    }
198
199    /**
200     * <h1>int serialDataAvail (int fd);</h1>
201     * 
202     * <p>
203     * Returns the number of characters available for reading, or -1 for any error condition, in
204     * which case errno will be set appropriately.
205     * </p>
206     * 
207     * @see <a
208     *      href="https://projects.drogon.net/raspberry-pi/wiringpi/serial-library/">https://projects.drogon.net/raspberry-pi/wiringpi/serial-library/</a>
209     * @param fd <p>
210     *            The file descriptor of the serial port.
211     *            </p>
212     * @return <p>
213     *         Returns the number of characters available for reading, or -1 for any error
214     *         condition, in which case errno will be set appropriately.
215     *         </p>
216     */
217    public static native int serialDataAvail(int fd);
218
219    /**
220     * <h1>int serialGetchar (int fd);</h1>
221     * 
222     * <p>
223     * Returns the next character available on the serial device. This call will block for up to 10
224     * seconds if no data is available (when it will return -1)
225     * </p>
226     * 
227     * @see <a
228     *      href="https://projects.drogon.net/raspberry-pi/wiringpi/serial-library/">https://projects.drogon.net/raspberry-pi/wiringpi/serial-library/</a>
229     * @param fd <p>
230     *            The file descriptor of the serial port.
231     *            </p>
232     * @return <p>
233     *         Returns the next character available on the serial device. This call will block for
234     *         up to 10 seconds if no data is available (when it will return -1)
235     *         </p>
236     */
237    public static native int serialGetchar(int fd);
238}
