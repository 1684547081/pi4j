017package com.pi4j.wiringpi;
018
019/**
020 * <h1>WiringPi Shift Library</h1>
021 * 
022 * <p>
023 * WiringPi includes a shift library which more or less mimics the one in the Arduino system. This
024 * allows you to shift 8-bit data values out of the Pi, or into the Pi from devices such as
025 * shift-registers (e.g. 74×595) and so-on, although it can also be used in some bit-banging
026 * scenarios.
027 * </p>
028 * 
029 * <p>
030 * Before using the Pi4J library, you need to ensure that the Java VM in configured with access to
031 * the following system libraries:
032 * <ul>
033 * <li>pi4j</li>
034 * <li>wiringPi</li>
035 * </ul>
036 * <blockquote> This library depends on the wiringPi native system library.</br> (developed by
037 * Gordon Henderson @ <a href="https://projects.drogon.net/">https://projects.drogon.net/</a>)
038 * </blockquote>
039 * </p>
040 * 
041 * @see <a href="http://www.pi4j.com/">http://www.pi4j.com/</a>
042 * @see <a
043 *      href="https://projects.drogon.net/raspberry-pi/wiringpi/shift-library/">https://projects.drogon.net/raspberry-pi/wiringpi/shift-library/</a>
044 * @author Robert Savage (<a
045 *         href="http://www.savagehomeautomation.com">http://www.savagehomeautomation.com</a>)
046 */
047public class Shift
048{
049    public static final int LSBFIRST = 0;
050    public static final int MSBFIRST = 1;
051
052    // Load the platform library
053    static
054    {
055        System.loadLibrary("pi4j");
056    }
057
058    /**
059     * <h1>uint8_t shiftIn (uint8_t dPin, uint8_t cPin, uint8_t order);</h1>
060     * 
061     * <p>
062     * This shifts an 8-bit data value in with the data appearing on the dPin and the clock being
063     * sent out on the cPin. Order is either LSBFIRST or MSBFIRST. The data is sampled after the
064     * cPin goes high. (So cPin high, sample data, cPin low, repeat for 8 bits) The 8-bit value is
065     * returned by the function.
066     * </p>
067     * 
068     * @see <a
069     *      href="https://projects.drogon.net/raspberry-pi/wiringpi/shift-library/">https://projects.drogon.net/raspberry-pi/wiringpi/shift-library/</a>
070     * 
071     * @param dPin
072     * @param cPin
073     * @param order
074     * @return <p>
075     *         The 8-bit shifted value is returned by the function.
076     *         </p>
077     */
078    public static native byte shiftIn(byte dPin, byte cPin, byte order);
079
080    /**
081     * <h1>void shiftOut (uint8_t dPin, uint8_t cPin, uint8_t order, uint8_t val);</h1>
082     * 
083     * <p>
084     * The shifts an 8-bit data value val out with the data being sent out on dPin and the clock
085     * being sent out on the cPin. order is as above. Data is clocked out on the rising or falling
086     * edge – ie. dPin is set, then cPin is taken high then low – repeated for the 8 bits.
087     * </p>
088     * 
089     * @see <a
090     *      href="https://projects.drogon.net/raspberry-pi/wiringpi/shift-library/">https://projects.drogon.net/raspberry-pi/wiringpi/shift-library/</a>
091     * 
092     * @param dPin
093     * @param cPin
094     * @param order
095     * @param val
096     */
097    public static native void shiftOut(byte dPin, byte cPin, byte order, byte val);
098}
