017package com.pi4j.wiringpi;
018
019/**
020 * <h1>WiringPi Software PWM Library</h1>
021 * 
022 * <p>
023 * WiringPi includes a software-driven PWM handler capable of outputting a PWM signal on any of the
024 * Raspberry Pi’s GPIO pins.
025 * </p>
026 * 
027 * <p>
028 * There are some limitations… To maintain a low CPU usage, the minimum pulse width is 100uS. That
029 * combined with the default suggested range of 100 gives a PWM frequency of 100Hz. You can lower
030 * the range to get a higher frequency, at the expense of resolution, or increase to get more
031 * resolution, but that will lower the frequency. If you change the pulse-width in the drive code,
032 * then be aware that at delays of less than 100uS wiringPi does it in a software loop, which means
033 * that CPU usage will rise dramatically, and controlling more than one pin will be almost
034 * impossible.
035 * </p>
036 * 
037 * <p>
038 * Also note that while the routines run themselves at a higher and real-time priority, Linux can
039 * still affect the accuracy of the generated signal.
040 * </p>
041 * 
042 * <p>
043 * However, within these limitations, control of a light/LED or a motor is very achievable.
044 * </p>
045 * 
046 * <p>
047 * <b> You must initialize wiringPi with one of wiringPiSetup() or wiringPiSetupGpio() functions
048 * beforehand. wiringPiSetupSys() is not fast enough, so you must run your programs with sudo. </b>
049 * </p>
050 * 
051 * <p>
052 * Before using the Pi4J library, you need to ensure that the Java VM in configured with access to
053 * the following system libraries:
054 * <ul>
055 * <li>pi4j</li>
056 * <li>wiringPi</li>
057 * <li>pthread</li>
058 * </ul>
059 * <blockquote> This library depends on the wiringPi native system library.</br> (developed by
060 * Gordon Henderson @ <a href="https://projects.drogon.net/">https://projects.drogon.net/</a>)
061 * </blockquote>
062 * </p>
063 * 
064 * @see <a href="http://www.pi4j.com/">http://www.pi4j.com/</a>
065 * @see <a
066 *      href="https://projects.drogon.net/raspberry-pi/wiringpi/software-pwm-library/">https://projects.drogon.net/raspberry-pi/wiringpi/software-pwm-library/</a>
067 * @author Robert Savage (<a
068 *         href="http://www.savagehomeautomation.com">http://www.savagehomeautomation.com</a>)
069 */
070public class SoftPwm
071{
072    // Load the platform library
073    static
074    {
075        System.loadLibrary("pi4j");
076    }
077
078    /**
079     * <h1>int softPwmCreate (int pin, int initialValue, int pwmRange);</h1>
080     * 
081     * <p>
082     * This creates a software controlled PWM pin. You can use any GPIO pin and the pin numbering
083     * will be that of the wiringPiSetup function you used. Use 100 for the pwmRange, then the value
084     * can be anything from 0 (off) to 100 (fully on) for the given pin.
085     * </p>
086     * 
087     * @see <a
088     *      href="https://projects.drogon.net/raspberry-pi/wiringpi/software-pwm-library/">https://projects.drogon.net/raspberry-pi/wiringpi/software-pwm-library/</a>
089     * 
090     * @param pin <p>
091     *            The GPIO pin to use as a PWM pin.
092     *            </p>
093     * @param value <p>
094     *            The value to initialize the PWM pin (between 0 <i>(off)</i> to 100 <i>(fully
095     *            on)</i>)
096     *            </p>
097     * @param range <p>
098     *            The maximum range. Use 100 for the pwmRange.
099     *            </p>
100     * @return <p>
101     *         The return value is 0 for success. Anything else and you should check the global
102     *         errno variable to see what went wrong.
103     *         </p>
104     */
105    public static native int softPwmCreate(int pin, int value, int range);
106
107    /**
108     * <h1>void softPwmWrite (int pin, int value);</h1>
109     * 
110     * <p>
111     * This updates the PWM value on the given pin. The value is checked to be in-range and pins
112     * that haven't previously been initialized via softPwmCreate will be silently ignored.
113     * </p>
114     * 
115     * @see <a
116     *      href="https://projects.drogon.net/raspberry-pi/wiringpi/software-pwm-library/">https://projects.drogon.net/raspberry-pi/wiringpi/software-pwm-library/</a>
117     * @param pin <p>
118     *            The GPIO pin to use as a PWM pin.
119     *            </p>
120     * @param value <p>
121     *            The value to initialize the PWM pin (between 0 <i>(off)</i> to 100 <i>(fully
122     *            on)</i>)
123     *            </p>
124     */
125    public static native void softPwmWrite(int pin, int value);
126}
