033/**
034 * <p>
035 * WiringPi includes a library which can make it easier to use the Raspberry Piâ€™s on-board SPI interface.
036 * </p>
037 *
038 * <p>
039 * Before you can use SPI interface, you may need to use the gpio utility to load the SPI drivers into the kernel:
040 *  > gpio load spi
041 *
042 * If you need a buffer size of greater than 4KB, then you can specify the size (in KB) on the command line:
043 *  > gpio load spi 100
044 *
045 * will allocate a 100KB buffer. (You should rarely need this though, the default is more than enough
046 * for most applications).
047 * </p>
048 * 
049 * <p>
050 * <blockquote> This library depends on the wiringPi native system library.</br> (developed by
051 * Gordon Henderson @ <a href="http://wiringpi.com/">http://wiringpi.com</a>)
052 * </blockquote>
053 * </p>
054 * 
055 * @see <a href="http://www.pi4j.com/">http://www.pi4j.com</a>
056 * @see <a
057 *      href="http://wiringpi.com/reference/spi-library/">http://wiringpi.com/reference/spi-library</a>
058 * @author Robert Savage (<a
059 *         href="http://www.savagehomeautomation.com">http://www.savagehomeautomation.com</a>)
060 */
061public class Spi {
062
063    public static int CHANNEL_0 = 0;
064    public static int CHANNEL_1 = 1;
065
066    public static int MODE_0 = 0;
067    public static int MODE_1 = 1;
068    public static int MODE_2 = 2;
069    public static int MODE_3 = 3;
070    
071    // private constructor 
072    private Spi()  {
073        // forbid object construction 
074    }
075    
076    static {
077        // Load the platform library
078        NativeLibraryLoader.load("libpi4j.so");
079    }
080
081    /**
082     * <p>wiringPiSPISetup:</p>
083     *
084     * <p>
085     * This is the way to initialise a channel (The Pi has 2 channels; 0 and 1). The speed parameter is an integer in
086     * the range 500,000 through 32,000,000 and represents the SPI clock speed in Hz.
087     * </p>
088     *
089     * <p>
090     * The returned value is the Linux file-descriptor for the device, or -1 on error. If an error has happened, you
091     * may use the standard errno global variable to see why.
092     * </p>
093     *
094     * @see <a
095     *      href="http://wiringpi.com/reference/spi-library/">http://wiringpi.com/reference/spi-library</a>
096     * @param channel SPI channel
097     * @param speed SPI speed
098     * @return return -1 on error
099     */
100    public static native int wiringPiSPISetup(int channel, int speed);
101
102
103    /**
104     * <p>wiringPiSPISetupMode:</p>
105     *
106     * <p>
107     * This is the way to initialise a channel (The Pi has 2 channels; 0 and 1). The speed parameter is an integer in
108     * the range 500,000 through 32,000,000 and represents the SPI clock speed in Hz.
109     * </p>
110     *
111     * <p>
112     * The returned value is the Linux file-descriptor for the device, or -1 on error. If an error has happened, you
113     * may use the standard errno global variable to see why.
114     * </p>
115     *
116     * @see <a
117     *      href="http://wiringpi.com/reference/spi-library/">http://wiringpi.com/reference/spi-library</a>
118     * @see <a
119     *      hfref="http://en.wikipedia.org/wiki/Serial_Peripheral_Interface_Bus#Mode_numbers">http://en.wikipedia.org/wiki/Serial_Peripheral_Interface_Bus#Mode_numbers</a>
120     * @param channel SPI channel
121     * @param speed SPI speed
122     * @param mode SPI mode (Mode is 0, 1, 2 or 3; see http://en.wikipedia.org/wiki/Serial_Peripheral_Interface_Bus#Mode_numbers)
123     * @return return -1 on error
124     */
125    public static native int wiringPiSPISetupMode (int channel, int speed, int mode);
126
127    /**
128     * <p>wiringPiSPIGetFd:</p>
129     * 
130     * <p>
131     * Return the file-descriptor for the given channel
132     * </p>
133     *
134     * @see <a
135     *      href="http://wiringpi.com/reference/spi-library/">http://wiringpi.com/reference/spi-library</a>
136     * @param channel SPI channel
137     * @return file-descriptor for the given channel
138     */
139    public static native int wiringPiSPIGetFd(int channel);
140
141    /**
142     * <p>wiringPiSPIDataRW:</p>
143     * 
144     * <p>
145     * This performs a simultaneous write/read transaction over the selected SPI bus. Data that was in your buffer is
146     * overwritten by data returned from the SPI bus.
147     * </p>
148     *
149     * <p>
150     * (ATTENTION: the 'data' argument can only be a maximum of 1024 characters.)
151     * </p>
152     *
153     * @see <a
154     *      href="http://wiringpi.com/reference/spi-library/">http://wiringpi.com/reference/spi-library</a>
155     * @param channel SPI channel</p>
156     * @param data string data payload
157     * @param len length of characters in string
158     * @return return -1 on error
159     */
160    public static native int wiringPiSPIDataRW(int channel, String data, int len);
161
162    /**
163     * <p>wiringPiSPIDataRW:</p>
164     *
165     * <p>
166     * This performs a simultaneous write/read transaction over the selected SPI bus. Data that was in your buffer is
167     * overwritten by data returned from the SPI bus.
168     * </p>
169     *
170     * <p>
171     * (ATTENTION: the 'data' argument can only be a maximum of 1024 characters.)
172     * </p>
173     *
174     * @see <a
175     *      href="http://wiringpi.com/reference/spi-library/">http://wiringpi.com/reference/spi-library</a>
176     * @param channel
177     *             SPI channel
178     * @param data
179     *             byte array data payload
180     * @param len
181     *             length of bytes in data array argument
182     * @return return -1 on error
183     */
184    public static native int wiringPiSPIDataRW(int channel, byte[] data, int len);
185
186
187    /**
188     * <p>wiringPiSPIDataRW:</p>
189     *
190     * <p>
191     * This performs a simultaneous write/read transaction over the selected SPI bus. Data that was in your buffer is
192     * overwritten by data returned from the SPI bus.
193     * </p>
194     *
195     * <p>
196     * (ATTENTION: the 'data' argument can only be a maximum of 1024 characters.)
197     * </p>
198     *
199     * @see <a
200     *      href="http://wiringpi.com/reference/spi-library/">http://wiringpi.com/reference/spi-library</a>
201     * @param channel
202     *             SPI channel
203     * @param data
204     *             byte array data payload
205     * @return return -1 on error
206     */
207    public static int wiringPiSPIDataRW(int channel, byte[] data){
208        return wiringPiSPIDataRW(channel, data, data.length);
209    }
210
211    /**
212     * <p>
213     * wiringPiSPIDataRW:
214     * </p>
215     *
216     * <p>
217     * This performs a simultaneous write/read transaction over the selected SPI bus. The data argument is passed into the wiringPI function as the argument and the output from Spi is returned by this
218     * method
219     * </p>
220     *
221     * @see <a href="http://wiringpi.com/reference/spi-library/">http://wiringpi.com/reference/spi-library</a>
222     * @param channel
223     *            SPI channel
224     * @param data
225     *            short array data payload. Note that wiringPi uses unsigned char for the data transmission. That is 8-bit. in other words values 0-255. So make sure the values in data do not exceed
226     *            this range, otherwise the numbers would overflow in the native code and unexpected results would yield
227     * @param len
228     *             length of bytes in data array argument
229     * @return return -1 on error
230     */
231    public static native int wiringPiSPIDataRW(int channel, short[] data, int len);
232
233    /**
234     * <p>
235     * wiringPiSPIDataRW:
236     * </p>
237     *
238     * <p>
239     * This performs a simultaneous write/read transaction over the selected SPI bus. The data argument is passed into the wiringPI function as the argument and the output from Spi is returned by this
240     * method
241     * </p>
242     *
243     * @see <a href="http://wiringpi.com/reference/spi-library/">http://wiringpi.com/reference/spi-library</a>

245     *            SPI channel
246     * @param data
247     *            short array data payload. Note that wiringPi uses unsigned char for the data transmission. That is 8-bit. in other words values 0-255. So make sure the values in data do not exceed
248     *            this range, otherwise the numbers would overflow in the native code and unexpected results would yield
249     * @return return -1 on error
250     */
251    public static int wiringPiSPIDataRW(int channel, short[] data){
252        return wiringPiSPIDataRW(channel, data, data.length);
253    }
