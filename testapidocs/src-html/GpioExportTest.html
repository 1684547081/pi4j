001import com.pi4j.wiringpi.GpioUtil;
002
003public class GpioExportTest
004{
005    public static void main(String args[]) throws InterruptedException
006    {
007        System.out.println("<--Pi4J--> GPIO export test program");
008        
009        // set GPIO 4 as the input trigger 
010        GpioUtil.export(24, GpioUtil.DIRECTION_IN);
011        System.out.println("GPIO 24 is now exported");
012
013        System.out.println("GPIO 24 is exported ? : "+ GpioUtil.isExported(24));
014        System.out.println("GPIO 24 edge ? : "+ GpioUtil.getDirection(24));
015        
016        System.out.println("GPIO 24 edge = 'none'");
017        GpioUtil.setEdgeDetection(24,GpioUtil.EDGE_NONE);        
018        System.out.println("GPIO 24 edge ? : "+ GpioUtil.getEdgeDetection(24));
019        Thread.sleep(5000);
020        
021        System.out.println("GPIO 24 edge = 'both'");
022        GpioUtil.setEdgeDetection(24,GpioUtil.EDGE_BOTH);        
023        System.out.println("GPIO 24 edge ? : "+ GpioUtil.getEdgeDetection(24));
024        Thread.sleep(5000);
025        
026        System.out.println("GPIO 24 edge = 'rising'");
027        GpioUtil.setEdgeDetection(24,GpioUtil.EDGE_RISING);        
028        System.out.println("GPIO 24 edge ? : "+ GpioUtil.getEdgeDetection(24));
029        Thread.sleep(5000);
030        
031        System.out.println("GPIO 24 edge = 'falling'");
032        GpioUtil.setEdgeDetection(24,GpioUtil.EDGE_FALLING);        
033        System.out.println("GPIO 24 edge ? : "+ GpioUtil.getEdgeDetection(24));
034        Thread.sleep(5000);
035
036        System.out.println("GPIO 24 direction = 'out'");
037        GpioUtil.setDirection(24, GpioUtil.DIRECTION_OUT);     
038        System.out.println("GPIO 24 direction ? : "+ GpioUtil.getDirection(24));
039        Thread.sleep(5000);
040
041        System.out.println("GPIO 24 direction = 'in'");
042        GpioUtil.setDirection(24, GpioUtil.DIRECTION_IN);     
043        System.out.println("GPIO 24 direction ? : "+ GpioUtil.getDirection(24));
044        Thread.sleep(5000);
045        
046        GpioUtil.unexport(24);
047        System.out.println("GPIO 24 is now unexported");
048        
049        System.out.println("GPIO 24 is exported ? : "+ GpioUtil.isExported(24));        
050    }
051}
