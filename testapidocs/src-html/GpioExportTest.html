027import com.pi4j.wiringpi.GpioUtil;
028
029public class GpioExportTest
030{
031    public static void main(String args[]) throws InterruptedException
032    {
033        System.out.println("<--Pi4J--> GPIO export test program");
034        
035        // set GPIO #07 as the input trigger 
036        GpioUtil.export(7, GpioUtil.DIRECTION_IN);
037        System.out.println("GPIO 07 is now exported");
038
039        System.out.println("GPIO 07 is exported ? : "+ GpioUtil.isExported(7));
040        System.out.println("GPIO 07 edge ? : "+ GpioUtil.getDirection(7));
041        
042        System.out.println("GPIO 07 edge = 'none'");
043        GpioUtil.setEdgeDetection(07,GpioUtil.EDGE_NONE);        
044        System.out.println("GPIO 07 edge ? : "+ GpioUtil.getEdgeDetection(7));
045        Thread.sleep(5000);
046        
047        System.out.println("GPIO 07 edge = 'both'");
048        GpioUtil.setEdgeDetection(7,GpioUtil.EDGE_BOTH);        
049        System.out.println("GPIO 07 edge ? : "+ GpioUtil.getEdgeDetection(7));
050        Thread.sleep(5000);
051        
052        System.out.println("GPIO 07 edge = 'rising'");
053        GpioUtil.setEdgeDetection(7,GpioUtil.EDGE_RISING);        
054        System.out.println("GPIO 07 edge ? : "+ GpioUtil.getEdgeDetection(7));
055        Thread.sleep(5000);
056        
057        System.out.println("GPIO 07 edge = 'falling'");
058        GpioUtil.setEdgeDetection(7,GpioUtil.EDGE_FALLING);        
059        System.out.println("GPIO 07 edge ? : "+ GpioUtil.getEdgeDetection(7));
060        Thread.sleep(5000);
061
062        System.out.println("GPIO 07 direction = 'out'");
063        GpioUtil.setDirection(7, GpioUtil.DIRECTION_OUT);     
064        System.out.println("GPIO 07 direction ? : "+ GpioUtil.getDirection(7));
065        Thread.sleep(5000);
066
067        System.out.println("GPIO 07 direction = 'in'");
068        GpioUtil.setDirection(7, GpioUtil.DIRECTION_IN);     
069        System.out.println("GPIO 07 direction ? : "+ GpioUtil.getDirection(7));
070        Thread.sleep(5000);
071        
072        GpioUtil.unexport(7);
073        System.out.println("GPIO 07 is now unexported");
074        
075        System.out.println("GPIO 07 is exported ? : "+ GpioUtil.isExported(7));        
076    }
077}
