027import java.util.concurrent.Callable;
028
029import com.pi4j.io.gpio.Gpio;
030import com.pi4j.io.gpio.GpioFactory;
031import com.pi4j.io.gpio.GpioPin;
032import com.pi4j.io.gpio.Pin;
033import com.pi4j.io.gpio.PinEdge;
034import com.pi4j.io.gpio.PinResistor;
035import com.pi4j.io.gpio.PinState;
036import com.pi4j.io.gpio.event.GpioListener;
037import com.pi4j.io.gpio.event.GpioPinStateChangeEvent;
038import com.pi4j.io.gpio.trigger.GpioCallbackTrigger;
039import com.pi4j.io.gpio.trigger.GpioInverseSyncStateTrigger;
040import com.pi4j.io.gpio.trigger.GpioPulseStateTrigger;
041import com.pi4j.io.gpio.trigger.GpioSetStateTrigger;
042
043public class GpioTest
044{
045    public static void main(String args[]) throws InterruptedException
046    {
047        System.out.println("<--Pi4J--> GPIO test program");
048
049        // create gpio listener & callback handler
050        GpioTestListener listener = new GpioTestListener();
051
052        // create gpio controller
053        Gpio gpio = GpioFactory.createInstance();
054
055        // setup gpio pin #07 as an input pin whose biased to ground and receives +3VDC to be
056        // triggered
057        GpioPin inputPin = gpio.provisionInputPin(Pin.GPIO_07, "Test-Input", PinEdge.BOTH,
058                                                  PinResistor.PULL_DOWN);
059
060        // add gpio listener
061        inputPin.addListener(listener);
062
063        // setup gpio pin #04, #05, #06 as output pins
064        GpioPin pinLed1 = gpio.provisionOuputPin(Pin.GPIO_04, "LED-One", PinState.LOW);
065        GpioPin pinLed2 = gpio.provisionOuputPin(Pin.GPIO_05, "LED-Two", PinState.HIGH);
066        GpioPin pinLed3 = gpio.provisionOuputPin(Pin.GPIO_06, "LED-Three", PinState.LOW);
067
068        // create gpio triggers
069        inputPin.addTrigger(new GpioSetStateTrigger(PinState.HIGH, pinLed1, PinState.HIGH));
070        inputPin.addTrigger(new GpioSetStateTrigger(PinState.LOW, pinLed1, PinState.LOW));
071        inputPin.addTrigger(new GpioInverseSyncStateTrigger(pinLed2));
072        inputPin.addTrigger(new GpioPulseStateTrigger(PinState.HIGH, pinLed3, 1000));
073        inputPin.addTrigger(new GpioCallbackTrigger(PinState.allStates(), new Callable<Void>()
074        {
075            public Void call() throws Exception
076            {
077                System.out.println(" --> GPIO TRIGGER CALLBACK RECEIVED ");
078                return null;
079            }
080        }));
081
082        // keep program alive
083        for (;;)
084        {
085            Thread.sleep(500);
086        }
087    }
088}
089
090class GpioTestListener implements GpioListener
091{
092    public void pinStateChanged(GpioPinStateChangeEvent event)
093    {
094        System.out.println(" --> GPIO PIN LISTENER STATE CHANGE: " + event.getPin() + " = "
095                + event.getState());
096    }
097}
