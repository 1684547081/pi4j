001import java.io.IOException;
002
003import com.pi4j.system.NetworkInfo;
004import com.pi4j.system.SystemInfo;
005
006
007public class SystemInfoTest
008{
009
010    /**
011     * @param args
012     * @throws InterruptedException 
013     * @throws IOException 
014     */
015    public static void main(String[] args) throws IOException, InterruptedException
016    {
017        System.out.println("Serial Number     :  " + SystemInfo.getSerial());
018        System.out.println("Processor         :  " + SystemInfo.getProcessor());
019        System.out.println("Hardware Revision :  " + SystemInfo.getRevision());
020        
021        System.out.println("Hostname          :  " + NetworkInfo.getHostname());
022        for(String ipAddress : NetworkInfo.getIPAddresses())
023            System.out.println("IP Addresses      :  " + ipAddress);
024        for(String fqdn : NetworkInfo.getFQDNs())
025            System.out.println("FQDN              :  " + fqdn);
026        for(String nameserver : NetworkInfo.getNameservers())
027            System.out.println("Nameserver        :  " + nameserver);    
028    }
029}
