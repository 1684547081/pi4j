001import com.pi4j.wiringpi.Gpio;
002
003public class WiringPiGpioTest
004{
005    // Simple sequencer data
006    // Triplets of LED, On/Off and delay
007
008    private static final int data[] =
009    { 0, 1, 1, 1, 1, 1, 0, 0, 0, 2, 1, 1, 1, 0, 0, 3, 1, 1, 2, 0, 0, 4, 1, 1, 3, 0, 0, 5, 1, 1, 4,
010            0, 0, 6, 1, 1, 5, 0, 0, 7, 1, 1, 6, 0, 1, 7, 0, 1,
011
012            0, 0,
013            1, // Extra delay
014
015            // Back again
016
017            7, 1, 1, 6, 1, 1, 7, 0, 0, 5, 1, 1, 6, 0, 0, 4, 1, 1, 5, 0, 0, 3, 1, 1, 4, 0, 0, 2, 1,
018            1, 3, 0, 0, 1, 1, 1, 2, 0, 0, 0, 1, 1, 1, 0, 1, 0, 0, 1,
019
020            0, 0, 1, // Extra delay
021
022            9, 9, 9, // End marker
023
024    };
025
026    public static void main(String args[]) throws InterruptedException
027    {
028        int pin;
029        int dataPtr;
030        int l, s, d;
031        
032        System.out.println("<--Pi4J--> GPIO test program");
033
034        // setup wiringPi
035        if (Gpio.wiringPiSetup() == -1)
036        {
037            System.out.println(" ==>> GPIO SETUP FAILED");
038            return;
039        }
040
041        // set GPIO 4 as the input trigger 
042        Gpio.pinMode (7, Gpio.INPUT) ;  
043        Gpio.pullUpDnControl(7, Gpio.PUD_DOWN);        
044
045        // set all other GPIO as outputs
046        for (pin = 0; pin < 7; ++pin)
047            Gpio.pinMode(pin, Gpio.OUTPUT);
048        
049        dataPtr = 0;
050        for (;;)
051        {
052            l = data[dataPtr++]; // LED
053            s = data[dataPtr++]; // State
054            d = data[dataPtr++]; // Duration (10ths)
055
056            if ((l + s + d) == 27)
057            {
058                dataPtr = 0;
059                continue;
060            }
061
062            Gpio.digitalWrite(l, s);
063            
064            if (Gpio.digitalRead(7) == 1) // Pressed as our switch shorts to ground
065                Gpio.delay(d * 10); // Faster!
066            else
067                Gpio.delay(d * 100);
068        }
069    }
070}
