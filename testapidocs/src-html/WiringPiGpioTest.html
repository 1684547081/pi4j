027import com.pi4j.wiringpi.Gpio;
028import com.pi4j.wiringpi.GpioUtil;
029
030public class WiringPiGpioTest
031{
032    // Simple sequencer data
033    // Triplets of LED, On/Off and delay
034
035    private static final int data[] =
036    { 0, 1, 1, 1, 1, 1, 0, 0, 0, 2, 1, 1, 1, 0, 0, 3, 1, 1, 2, 0, 0, 4, 1, 1, 3, 0, 0, 5, 1, 1, 4,
037            0, 0, 6, 1, 1, 5, 0, 0, 7, 1, 1, 6, 0, 1, 7, 0, 1,
038
039            0, 0,
040            1, // Extra delay
041
042            // Back again
043
044            7, 1, 1, 6, 1, 1, 7, 0, 0, 5, 1, 1, 6, 0, 0, 4, 1, 1, 5, 0, 0, 3, 1, 1, 4, 0, 0, 2, 1,
045            1, 3, 0, 0, 1, 1, 1, 2, 0, 0, 0, 1, 1, 1, 0, 1, 0, 0, 1,
046
047            0, 0, 1, // Extra delay
048
049            9, 9, 9, // End marker
050
051    };
052
053    public static void main(String args[]) throws InterruptedException
054    {
055        int pin;
056        int dataPtr;
057        int l, s, d;
058        
059        System.out.println("<--Pi4J--> GPIO test program");
060
061        // setup wiringPi
062        if (Gpio.wiringPiSetup() == -1)
063        {
064            System.out.println(" ==>> GPIO SETUP FAILED");
065            return;
066        }
067
068        // set GPIO 4 as the input trigger 
069        GpioUtil.export(7, GpioUtil.DIRECTION_IN);
070        GpioUtil.setEdgeDetection(7, GpioUtil.EDGE_BOTH);
071        Gpio.pinMode (7, Gpio.INPUT) ;  
072        Gpio.pullUpDnControl(7, Gpio.PUD_DOWN);        
073
074        // set all other GPIO as outputs
075        for (pin = 0; pin < 7; ++pin)
076        {
077            // export all the GPIO pins that we will be using
078            GpioUtil.export(pin, GpioUtil.DIRECTION_OUT);            
079            Gpio.pinMode(pin, Gpio.OUTPUT);
080        }
081        
082        dataPtr = 0;
083        for (;;)
084        {
085            l = data[dataPtr++]; // LED
086            s = data[dataPtr++]; // State
087            d = data[dataPtr++]; // Duration (10ths)
088
089            if ((l + s + d) == 27)
090            {
091                dataPtr = 0;
092                continue;
093            }
094
095            Gpio.digitalWrite(l, s);
096            
097            if (Gpio.digitalRead(7) == 1) // Pressed as our switch shorts to ground
098                Gpio.delay(d * 10); // Faster!
099            else
100                Gpio.delay(d * 100);
101        }
102    }
103}
