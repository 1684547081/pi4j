002package com.pi4j.example;
003
004import com.pi4j.io.gpio.Gpio;
005import com.pi4j.io.gpio.GpioFactory;
006import com.pi4j.io.gpio.GpioPin;
007import com.pi4j.io.gpio.Pin;
008import com.pi4j.io.gpio.PinState;
009
010/**
011 * This example code demonstrates how to perform simple state
012 * control of a GPIO pin on the Raspberry Pi.  
013 * 
014 * @author Robert Savage
015 */
016public class ControlGpioExample
017{
018    public static void main(String[] args) throws InterruptedException
019    {
020        // create gpio controller
021        Gpio gpio = GpioFactory.createInstance();
022        
023        // provision gpio pin #4 as an output pin and turn on
024        GpioPin pin = gpio.provisionOuputPin(Pin.GPIO_04, "MyLED", PinState.HIGH);
025        
026        Thread.sleep(5000);
027        
028        // turn off gpio pin #4
029        pin.low();
030
031        Thread.sleep(5000);
032
033        // toggle the current state of gpio pin #4 (should turn on)
034        pin.toggle();
035
036        Thread.sleep(5000);
037
038        // toggle the current state of gpio pin #4  (should turn off)
039        pin.toggle();
040        
041        Thread.sleep(5000);
042
043        // turn on gpio pin #4 for 1 second and then off
044        pin.pulse(1000);
045    }
046}
