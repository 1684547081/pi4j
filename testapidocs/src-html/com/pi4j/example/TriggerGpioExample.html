002package com.pi4j.example;
003
004import java.util.concurrent.Callable;
005
006import com.pi4j.io.gpio.Gpio;
007import com.pi4j.io.gpio.GpioFactory;
008import com.pi4j.io.gpio.GpioPin;
009import com.pi4j.io.gpio.Pin;
010import com.pi4j.io.gpio.PinEdge;
011import com.pi4j.io.gpio.PinResistor;
012import com.pi4j.io.gpio.PinState;
013import com.pi4j.io.gpio.trigger.GpioCallbackTrigger;
014import com.pi4j.io.gpio.trigger.GpioPulseStateTrigger;
015import com.pi4j.io.gpio.trigger.GpioSetStateTrigger;
016import com.pi4j.io.gpio.trigger.GpioSyncStateTrigger;
017
018/**
019 * This example code demonstrates how to setup simple triggers for GPIO pins on the Raspberry Pi.
020 * 
021 * @author Robert Savage
022 */
023public class TriggerGpioExample
024{
025    public static void main(String[] args) throws InterruptedException
026    {
027        // create gpio controller
028        Gpio gpio = GpioFactory.createInstance();
029
030        // provision gpio pin #4 as an input pin with its internal pull down resistor enabled
031        // (configure pin edge to both rising and falling to get notified for HIGH and LOW state
032        // changes)
033        GpioPin myButton = gpio.provisionInputPin(Pin.GPIO_04, "MyButton", PinEdge.BOTH, PinResistor.PULL_DOWN);
034
035        // setup gpio pins #21,22,23 as an output pins and make sure they are all LOW at startup
036        GpioPin myLed[] = { 
037                gpio.provisionOuputPin(Pin.GPIO_21, "LED #1", PinState.LOW), 
038                gpio.provisionOuputPin(Pin.GPIO_22, "LED #2", PinState.LOW),
039                gpio.provisionOuputPin(Pin.GPIO_23, "LED #3", PinState.LOW) };
040        
041        // create a gpio control trigger on gpio pin #4; when #4 goes HIGH, also set pin #21 to HIGH
042        myButton.addTrigger(new GpioSetStateTrigger(PinState.HIGH, myLed[0], PinState.HIGH));
043
044        // create a gpio control trigger on gpio pin #4; when #4 goes LOW, also set pin #21 to LOW
045        myButton.addTrigger(new GpioSetStateTrigger(PinState.LOW, myLed[0], PinState.LOW));
046        
047        // create a gpio synchronization trigger on gpio pin#4; when #4 changes, also set pin #22 to
048        // same state
049        myButton.addTrigger(new GpioSyncStateTrigger(myLed[1]));
050
051        // create a gpio pulse trigger on gpio pin#4; when #4 goes HIGH, also pulse pin #23 to the
052        // HIGH state for 1 second
053        myButton.addTrigger(new GpioPulseStateTrigger(PinState.HIGH, myLed[2], 1000));
054        
055        // create a gpio callback trigger on gpio pin#4; when #4 changes state, perform a callback
056        // invocation on the user defined 'Callable' class instance
057        myButton.addTrigger(new GpioCallbackTrigger(new Callable<Void>()
058                {
059                    public Void call() throws Exception
060                    {
061                        System.out.println(" --> GPIO TRIGGER CALLBACK RECEIVED ");
062                        return null;
063                    }
064                }));
065
066        // keep program running until user aborts (CTRL-C)
067        for (;;)
068        {
069            Thread.sleep(500);
070        }
071    }
072}
