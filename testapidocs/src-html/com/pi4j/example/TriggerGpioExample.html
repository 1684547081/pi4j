002package com.pi4j.example;
003
004import java.util.concurrent.Callable;
005
006import com.pi4j.io.gpio.Gpio;
007import com.pi4j.io.gpio.GpioFactory;
008import com.pi4j.io.gpio.GpioPin;
009import com.pi4j.io.gpio.GpioPinDirection;
010import com.pi4j.io.gpio.GpioPinEdge;
011import com.pi4j.io.gpio.GpioPinResistor;
012import com.pi4j.io.gpio.GpioPinState;
013import com.pi4j.io.gpio.trigger.GpioCallbackTrigger;
014import com.pi4j.io.gpio.trigger.GpioPulseStateTrigger;
015import com.pi4j.io.gpio.trigger.GpioSetStateTrigger;
016import com.pi4j.io.gpio.trigger.GpioSyncStateTrigger;
017
018/**
019 * This example code demonstrates how to setup simple triggers for GPIO pins on the Raspberry Pi.
020 * 
021 * @author Robert Savage
022 */
023public class TriggerGpioExample
024{
025    public static void main(String[] args) throws InterruptedException
026    {
027        // create gpio controller
028        Gpio gpio = GpioFactory.createInstance();
029
030        // create a gpio control trigger on gpio pin#4; when #4 goes HIGH, also set pin #21 to HIGH
031        GpioSetStateTrigger triggerOn = new GpioSetStateTrigger(GpioPin.GPIO_04, GpioPinState.HIGH,
032                GpioPin.GPIO_21, GpioPinState.HIGH);
033
034        // create a gpio control trigger on gpio pin#4; when #4 goes LOW, also set pin #21 to LOW
035        GpioSetStateTrigger triggerOff = new GpioSetStateTrigger(GpioPin.GPIO_04, GpioPinState.LOW,
036                GpioPin.GPIO_21, GpioPinState.LOW);
037
038        // create a gpio synchronization trigger on gpio pin#4; when #4 changes, also set pin #22 to
039        // same state
040        GpioSyncStateTrigger triggerSync = new GpioSyncStateTrigger(GpioPin.GPIO_04,
041                GpioPinState.allStates(), GpioPin.GPIO_22);
042
043        // create a gpio pulse trigger on gpio pin#4; when #4 goes HIGH, also pulse pin #23 to the
044        // HIGH state for 1 second
045        GpioPulseStateTrigger triggerPulse = new GpioPulseStateTrigger(GpioPin.GPIO_04,
046                GpioPinState.HIGH, GpioPin.GPIO_23, 1000);
047
048        // create a gpio callback trigger on gpio pin#4; when #4 changes state, perform a callback
049        // invocation on the user defined 'Callable' class instance
050        GpioCallbackTrigger triggerCallback = new GpioCallbackTrigger(GpioPin.GPIO_04,
051                GpioPinState.allStates(), new Callable<Void>()
052                {
053                    public Void call() throws Exception
054                    {
055                        System.out.println(" --> GPIO TRIGGER CALLBACK RECEIVED ");
056                        return null;
057                    }
058                });
059
060        // register all the gpio triggers with the gpio controller
061        gpio.addTrigger(triggerOn);
062        gpio.addTrigger(triggerOff);
063        gpio.addTrigger(triggerSync);
064        gpio.addTrigger(triggerCallback);
065        gpio.addTrigger(triggerPulse);
066
067        // setup gpio pin #4 as an input pin with its internal pull down resistor enabled
068        // (configure pin edge to both rising and falling to get notified for HIGH and LOW state
069        // changes)
070        gpio.setup(GpioPin.GPIO_04, GpioPinDirection.IN, GpioPinEdge.BOTH,
071                   GpioPinResistor.PULL_DOWN);
072
073        // setup gpio pins #21,22,23 as an output pins and make sure they are all LOW at startup
074        GpioPin[] pins = { GpioPin.GPIO_21, GpioPin.GPIO_22, GpioPin.GPIO_23 };
075        gpio.setup(pins, GpioPinDirection.OUT);
076        gpio.setLow(pins);
077
078        // keep program running until user aborts (CTRL-C)
079        for (;;)
080        {
081            Thread.sleep(500);
082        }
083    }
084}
