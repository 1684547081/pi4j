032import java.util.concurrent.Callable;
033
034import com.pi4j.io.gpio.Gpio;
035import com.pi4j.io.gpio.GpioFactory;
036import com.pi4j.io.gpio.GpioPin;
037import com.pi4j.io.gpio.Pin;
038import com.pi4j.io.gpio.PinEdge;
039import com.pi4j.io.gpio.PinResistor;
040import com.pi4j.io.gpio.PinState;
041import com.pi4j.io.gpio.trigger.GpioCallbackTrigger;
042import com.pi4j.io.gpio.trigger.GpioPulseStateTrigger;
043import com.pi4j.io.gpio.trigger.GpioSetStateTrigger;
044import com.pi4j.io.gpio.trigger.GpioSyncStateTrigger;
045
046/**
047 * This example code demonstrates how to setup simple triggers for GPIO pins on the Raspberry Pi.
048 * 
049 * @author Robert Savage
050 */
051public class TriggerGpioExample
052{
053    public static void main(String[] args) throws InterruptedException
054    {
055        // create gpio controller
056        Gpio gpio = GpioFactory.createInstance();
057
058        // provision gpio pin #4 as an input pin with its internal pull down resistor enabled
059        // (configure pin edge to both rising and falling to get notified for HIGH and LOW state
060        // changes)
061        GpioPin myButton = gpio.provisionInputPin(Pin.GPIO_04, "MyButton", PinEdge.BOTH, PinResistor.PULL_DOWN);
062
063        // setup gpio pins #21,22,23 as an output pins and make sure they are all LOW at startup
064        GpioPin myLed[] = { 
065                gpio.provisionOuputPin(Pin.GPIO_21, "LED #1", PinState.LOW), 
066                gpio.provisionOuputPin(Pin.GPIO_22, "LED #2", PinState.LOW),
067                gpio.provisionOuputPin(Pin.GPIO_23, "LED #3", PinState.LOW) };
068        
069        // create a gpio control trigger on gpio pin #4; when #4 goes HIGH, also set pin #21 to HIGH
070        myButton.addTrigger(new GpioSetStateTrigger(PinState.HIGH, myLed[0], PinState.HIGH));
071
072        // create a gpio control trigger on gpio pin #4; when #4 goes LOW, also set pin #21 to LOW
073        myButton.addTrigger(new GpioSetStateTrigger(PinState.LOW, myLed[0], PinState.LOW));
074        
075        // create a gpio synchronization trigger on gpio pin#4; when #4 changes, also set pin #22 to
076        // same state
077        myButton.addTrigger(new GpioSyncStateTrigger(myLed[1]));
078
079        // create a gpio pulse trigger on gpio pin#4; when #4 goes HIGH, also pulse pin #23 to the
080        // HIGH state for 1 second
081        myButton.addTrigger(new GpioPulseStateTrigger(PinState.HIGH, myLed[2], 1000));
082        
083        // create a gpio callback trigger on gpio pin#4; when #4 changes state, perform a callback
084        // invocation on the user defined 'Callable' class instance
085        myButton.addTrigger(new GpioCallbackTrigger(new Callable<Void>()
086                {
087                    public Void call() throws Exception
088                    {
089                        System.out.println(" --> GPIO TRIGGER CALLBACK RECEIVED ");
090                        return null;
091                    }
092                }));
093
094        // keep program running until user aborts (CTRL-C)
095        for (;;)
096        {
097            Thread.sleep(500);
098        }
099    }
100}
