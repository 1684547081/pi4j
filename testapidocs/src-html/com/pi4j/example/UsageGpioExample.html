050/**
051 * This example code demonstrates how to setup simple triggers for GPIO pins on the Raspberry Pi.
052 * 
053 * @author Robert Savage
054 */
055@SuppressWarnings("unused")
056public class UsageGpioExample
057{
058    public static void main(String[] args) throws InterruptedException
059    {
060        // START SNIPPET: usage-create-controller-snippet
061        // create gpio controller instance
062        Gpio gpio = GpioFactory.createInstance();
063        // END SNIPPET: usage-create-controller-snippet
064
065        // START SNIPPET: usage-provision-input-pin-snippet
066        // provision gpio pin #02 as an input pin with its internal pull down resistor enabled
067        // (configure pin edge to both rising and falling to get notified for HIGH and LOW state
068        // changes)
069        GpioPin myButton = gpio.provisionInputPin(Pin.GPIO_02,            // PIN NUMBER
070                                                  "MyButton",             // PIN FRIENDLY NAME
071                                                  PinEdge.BOTH,           // PIN EDGE DETECTION
072                                                  PinResistor.PULL_DOWN); // PIN RESISTANCE
073
074        // END SNIPPET: usage-provision-input-pin-snippet
075        // START SNIPPET: usage-provision-output-pin-snippet
076        // provision gpio pins #04 as an output pin and make sure is is set to LOW at startup
077        GpioPin myLed = gpio.provisionOuputPin(Pin.GPIO_04,   // PIN NUMBER
078                                               "My LED",      // PIN FRIENDLY NAME
079                                               PinState.LOW); // PIN STARTUP STATE
080        // END SNIPPET: usage-provision-output-pin-snippet
081
082        // START SNIPPET: usage-control-pin-snippet
083        // explicitly set a state on the pin object
084        myLed.setState(PinState.HIGH);
085
086        // use convenience wrapper method to set state on the pin object
087        myLed.low();
088        myLed.high();
089
090        // use toggle method to apply inverse state on the pin object
091        myLed.toggle();
092
093        // use pulse method to set the pin to the HIGH state for
094        // an explicit length of time in milliseconds
095        myLed.pulse(1000);
096        // END SNIPPET: usage-control-pin-snippet
097
098        // START SNIPPET: usage-read-pin-snippet
099        // get explicit state enumeration for the GPIO pin associated with the button
100        PinState myButtonState = myButton.getState();
101
102        // use convenience wrapper method to interrogate the button state
103        boolean buttonPressed = myButton.isHigh();
104        // END SNIPPET: usage-read-pin-snippet
105
106        // START SNIPPET: usage-register-listener-snippet
107        // create and register gpio pin listener
108        myButton.addListener(new GpioExampleListener());
109        // END SNIPPET: usage-register-listener-snippet
110
111        // START SNIPPET: usage-trigger-snippet
112        // create a gpio synchronization trigger on the input pin
113        // when the input state changes, also set LED controlling gpio pin to same state
114        myButton.addTrigger(new GpioSyncStateTrigger(myLed));
115        // END SNIPPET: usage-trigger-snippet
116
117        // keep program running until user aborts (CTRL-C)
118        for (;;)
119        {
120            Thread.sleep(500);
121        }
122    }
123
124 //START SNIPPET: usage-listener-snippet    
125    class GpioUsageExampleListener implements GpioListener
126    {
127        public void pinStateChanged(GpioPinStateChangeEvent event)
128        {
129            System.out.println(" --> GPIO PIN STATE CHANGE: " + event.getPin() + " = "
130                    + event.getState());
131        }
132    }
133 // END SNIPPET: usage-listener-snippet
134}
