10  /**
11   * This example code demonstrates how to perform simple state
12   * control of a GPIO pin on the Raspberry Pi.  
13   * 
14   * @author Robert Savage
15   */
16  public class ControlGpioExample
17  {
18      public static void main(String[] args) throws InterruptedException
19      {
20          // create gpio controller
21          Gpio gpio = GpioFactory.createInstance();
22          
23          // setup gpio pin #4 as an output pin  
24          gpio.setup(GpioPin.GPIO_04, GpioPinDirection.OUT);
25          
26          // turn on gpio pin #4
27          gpio.setState(GpioPin.GPIO_04, GpioPinState.HIGH);
28  
29          Thread.sleep(5000);
30          
31          // turn off gpio pin #4
32          gpio.setState(GpioPin.GPIO_04, GpioPinState.LOW);
33  
34          Thread.sleep(5000);
35  
36          // toggle the current state of gpio pin #4 (should turn on)
37          gpio.toggleState(GpioPin.GPIO_04);
38  
39          Thread.sleep(5000);
40  
41          // toggle the current state of gpio pin #4  (should turn off)
42          gpio.toggleState(GpioPin.GPIO_04);
43          
44          Thread.sleep(5000);
45  
46          // turn on gpio pin #4 for 1 second and then off
47          gpio.pulse(GpioPin.GPIO_04, 1000);        
48      }
49  }
