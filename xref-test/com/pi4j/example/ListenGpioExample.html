13  /**
14   * This example code demonstrates how to setup a listener
15   * for GPIO pin state changes on the Raspberry Pi.  
16   * 
17   * @author Robert Savage
18   */
19  public class ListenGpioExample
20  {
21      public static void main(String args[]) throws InterruptedException
22      {
23          // create gpio controller
24          Gpio gpio = GpioFactory.createInstance();
25  
26          // provision gpio pin #4 as an input pin with its internal pull down resistor enabled
27          // (configure pin edge to both rising and falling to get notified for HIGH and LOW state changes)
28          GpioPin myButton = gpio.provisionInputPin(Pin.GPIO_04, "MyButton", PinEdge.BOTH, PinResistor.PULL_DOWN);
29  
30          // create and register gpio pin listener
31          myButton.addListener(new GpioExampleListener());
32  
33          // keep program running until user aborts (CTRL-C)
34          for (;;)
35          {
36              Thread.sleep(500);
37          }
38      }
39  }

41  /**
42   * This class implements the GPIO listener interface
43   * with the callback method for event notifications
44   * when GPIO pin states change.
45   * 
46   * @see GpioListener
47   * @author Robert Savage
48   */
49  class GpioExampleListener implements GpioListener
50  {
51      public void pinStateChanged(GpioPinStateChangeEvent event)
52      {
53          System.out.println(" --> GPIO PIN STATE CHANGE: " + event.getPin() + " = "
54                  + event.getState());
55      }
56  }
