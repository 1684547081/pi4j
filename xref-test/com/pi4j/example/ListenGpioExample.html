32  import com.pi4j.io.gpio.Gpio;
33  import com.pi4j.io.gpio.GpioFactory;
34  import com.pi4j.io.gpio.GpioPin;
35  import com.pi4j.io.gpio.Pin;

37  import com.pi4j.io.gpio.PinResistor;
38  import com.pi4j.io.gpio.event.GpioListener;
39  import com.pi4j.io.gpio.event.GpioPinStateChangeEvent;

41  /**
42   * This example code demonstrates how to setup a listener
43   * for GPIO pin state changes on the Raspberry Pi.  
44   * 
45   * @author Robert Savage
46   */
47  public class ListenGpioExample
48  {
49      public static void main(String args[]) throws InterruptedException
50      {
51          System.out.println("<--Pi4J--> GPIO Listen Example ... started.");
52          
53          // create gpio controller
54          Gpio gpio = GpioFactory.createInstance();
55  
56          // provision gpio pin #02 as an input pin with its internal pull down resistor enabled
57          // (configure pin edge to both rising and falling to get notified for HIGH and LOW state changes)
58          GpioPin myButton = gpio.provisionInputPin(Pin.GPIO_02, "MyButton", PinEdge.BOTH, PinResistor.PULL_DOWN);
59  
60          // create and register gpio pin listener
61          myButton.addListener(new GpioExampleListener());
62          
63          System.out.println(" ... complete the GPIO #02 circuit and see the listener feedback here in the console.");
64  
65          // keep program running until user aborts (CTRL-C)
66          for (;;)
67          {
68              Thread.sleep(500);
69          }
70      }
71  }

73  /**
74   * This class implements the GPIO listener interface
75   * with the callback method for event notifications
76   * when GPIO pin states change.
77   * 
78   * @see GpioListener
79   * @author Robert Savage
80   */
81  class GpioExampleListener implements GpioListener
82  {
83      public void pinStateChanged(GpioPinStateChangeEvent event)
84      {
85          System.out.println(" --> GPIO PIN STATE CHANGE: " + event.getPin() + " = "
86                  + event.getState());
87      }
88  }
