32  import java.util.Date;
33  
34  import com.pi4j.io.serial.Serial;
35  import com.pi4j.io.serial.SerialDataEvent;
36  import com.pi4j.io.serial.SerialDataListener;
37  import com.pi4j.io.serial.SerialFactory;
38  
39  /**
40   * This example code demonstrates how to perform serial communications using the Raspberry Pi.
41   * 
42   * @author Robert Savage
43   */
44  public class SerialExample
45  {
46      public static void main(String args[]) throws InterruptedException
47      {
48          // !! ATTENTION !!
49          // By default, the serial port is configured as a console port 
50          // for interacting with the Linux OS shell.  If you want to use 
51          // the serial port in a software program, you must disable the 
52          // OS from using this port.  Please see this blog article by  
53          // Clayton Smith for step-by-step instructions on how to disable 
54          // the OS console for this port:
55          // http://www.irrational.net/2012/04/19/using-the-raspberry-pis-serial-port/
56                  
57          System.out.println("<--Pi4J--> Serial Communication Example ... started.");
58          System.out.println(" ... connect using settings: 38400, N, 8, 1.");
59          System.out.println(" ... data received on serial port should be displayed below.");
60          
61          // create serial data listener
62          SerialExampleListener listener = new SerialExampleListener();
63  
64          // create an instance of the serial communications class
65          Serial serial = SerialFactory.createInstance();
66  
67          // add/register the serial data listener
68          serial.addListener(listener);
69          
70          // open the default serial port provided on the GPIO header
71          int ret = serial.open(Serial.DEFAULT_COM_PORT, 38400);
72          if (ret == -1)
73          {
74              System.out.println(" ==>> SERIAL SETUP FAILED");
75              return;
76          }
77  
78          // continuous loop to keep the program running until the user terminates the program
79          for (;;)
80          {
81              // write a formatted string to the serial transmit buffer
82              serial.write("CURRENT TIME: %s", new Date().toString());
83  
84              // write a individual bytes to the serial transmit buffer
85              serial.write((byte) 13);
86              serial.write((byte) 10);
87  
88              // write a simple string to the serial transmit buffer
89              serial.write("Second Line");
90  
91              // write a individual characters to the serial transmit buffer
92              serial.write('\r');
93              serial.write('\n');
94  
95              // write a string terminating with CR+LF to the serial transmit buffer
96              serial.writeln("Third Line");
97  
98              // wait 1 second before continuing
99              Thread.sleep(1000);
100         }
101     }
102 }

104 /**
105  * This class implements the serial data listener interface with the callback method for event
106  * notifications when data is received on the serial port.
107  * 
108  * @see SerialDataListener
109  * @author Robert Savage
110  */
111 class SerialExampleListener implements SerialDataListener
112 {
113     public void dataReceived(SerialDataEvent event)
114     {
115         // print out the data received to the console
116         System.out.print(event.getData());
117     }
118 }
