13   * @author Robert Savage

15  public class SystemInfoExample
16  {
17      public static void main(String[] args) throws InterruptedException, IOException
18      {
19          // display a few of the available system information properties
20          System.out.println("Serial Number     :  " + SystemInfo.getSerial());
21          System.out.println("CPU Revision      :  " + SystemInfo.getCpuRevision());
22          System.out.println("CPU Architecture  :  " + SystemInfo.getCpuArchitecture());
23          System.out.println("CPU Part          :  " + SystemInfo.getCpuPart());
24          System.out.println("MIPS              :  " + SystemInfo.getBogoMIPS());
25          System.out.println("Processor         :  " + SystemInfo.getProcessor());
26          System.out.println("Hardware Revision :  " + SystemInfo.getRevision());
27  
28          // display some of the network information
29          System.out.println("Hostname          :  " + NetworkInfo.getHostname());
30          for (String ipAddress : NetworkInfo.getIPAddresses())
31              System.out.println("IP Addresses      :  " + ipAddress);
32          for (String fqdn : NetworkInfo.getFQDNs())
33              System.out.println("FQDN              :  " + fqdn);
34          for (String nameserver : NetworkInfo.getNameservers())
35              System.out.println("Nameserver        :  " + nameserver);
36      }
