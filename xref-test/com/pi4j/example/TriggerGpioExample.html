2   package com.pi4j.example;

4   import java.util.concurrent.Callable;
5   
6   import com.pi4j.io.gpio.Gpio;
7   import com.pi4j.io.gpio.GpioFactory;
8   import com.pi4j.io.gpio.GpioPin;
9   import com.pi4j.io.gpio.Pin;
10  import com.pi4j.io.gpio.PinEdge;
11  import com.pi4j.io.gpio.PinResistor;
12  import com.pi4j.io.gpio.PinState;
13  import com.pi4j.io.gpio.trigger.GpioCallbackTrigger;
14  import com.pi4j.io.gpio.trigger.GpioPulseStateTrigger;
15  import com.pi4j.io.gpio.trigger.GpioSetStateTrigger;
16  import com.pi4j.io.gpio.trigger.GpioSyncStateTrigger;
17  
18  /**
19   * This example code demonstrates how to setup simple triggers for GPIO pins on the Raspberry Pi.
20   * 
21   * @author Robert Savage
22   */
23  public class TriggerGpioExample
24  {
25      public static void main(String[] args) throws InterruptedException
26      {
27          // create gpio controller
28          Gpio gpio = GpioFactory.createInstance();
29  
30          // provision gpio pin #4 as an input pin with its internal pull down resistor enabled
31          // (configure pin edge to both rising and falling to get notified for HIGH and LOW state
32          // changes)
33          GpioPin myButton = gpio.provisionInputPin(Pin.GPIO_04, "MyButton", PinEdge.BOTH, PinResistor.PULL_DOWN);
34  
35          // setup gpio pins #21,22,23 as an output pins and make sure they are all LOW at startup
36          GpioPin myLed[] = { 
37                  gpio.provisionOuputPin(Pin.GPIO_21, "LED #1", PinState.LOW), 
38                  gpio.provisionOuputPin(Pin.GPIO_22, "LED #2", PinState.LOW),
39                  gpio.provisionOuputPin(Pin.GPIO_23, "LED #3", PinState.LOW) };
40          
41          // create a gpio control trigger on gpio pin #4; when #4 goes HIGH, also set pin #21 to HIGH
42          myButton.addTrigger(new GpioSetStateTrigger(PinState.HIGH, myLed[0], PinState.HIGH));
43  
44          // create a gpio control trigger on gpio pin #4; when #4 goes LOW, also set pin #21 to LOW
45          myButton.addTrigger(new GpioSetStateTrigger(PinState.LOW, myLed[0], PinState.LOW));
46          
47          // create a gpio synchronization trigger on gpio pin#4; when #4 changes, also set pin #22 to
48          // same state
49          myButton.addTrigger(new GpioSyncStateTrigger(myLed[1]));
50  
51          // create a gpio pulse trigger on gpio pin#4; when #4 goes HIGH, also pulse pin #23 to the
52          // HIGH state for 1 second
53          myButton.addTrigger(new GpioPulseStateTrigger(PinState.HIGH, myLed[2], 1000));
54          
55          // create a gpio callback trigger on gpio pin#4; when #4 changes state, perform a callback
56          // invocation on the user defined 'Callable' class instance
57          myButton.addTrigger(new GpioCallbackTrigger(new Callable<Void>()
58                  {
59                      public Void call() throws Exception
60                      {
61                          System.out.println(" --> GPIO TRIGGER CALLBACK RECEIVED ");
62                          return null;
63                      }
64                  }));
65  
66          // keep program running until user aborts (CTRL-C)
67          for (;;)
68          {
69              Thread.sleep(500);
70          }
71      }
72  }
