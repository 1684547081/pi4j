32  import java.util.concurrent.Callable;
33  
34  import com.pi4j.io.gpio.Gpio;
35  import com.pi4j.io.gpio.GpioFactory;
36  import com.pi4j.io.gpio.GpioPin;
37  import com.pi4j.io.gpio.Pin;
38  import com.pi4j.io.gpio.PinEdge;
39  import com.pi4j.io.gpio.PinResistor;
40  import com.pi4j.io.gpio.PinState;
41  import com.pi4j.io.gpio.trigger.GpioCallbackTrigger;
42  import com.pi4j.io.gpio.trigger.GpioPulseStateTrigger;
43  import com.pi4j.io.gpio.trigger.GpioSetStateTrigger;
44  import com.pi4j.io.gpio.trigger.GpioSyncStateTrigger;
45  
46  /**
47   * This example code demonstrates how to setup simple triggers for GPIO pins on the Raspberry Pi.
48   * 
49   * @author Robert Savage
50   */
51  public class TriggerGpioExample
52  {
53      public static void main(String[] args) throws InterruptedException
54      {
55          // create gpio controller
56          Gpio gpio = GpioFactory.createInstance();
57  
58          // provision gpio pin #02 as an input pin with its internal pull down resistor enabled
59          // (configure pin edge to both rising and falling to get notified for HIGH and LOW state changes)
60          GpioPin myButton = gpio.provisionInputPin(Pin.GPIO_02, 
61                                                    "MyButton", 
62                                                    PinEdge.BOTH,
63                                                    PinResistor.PULL_DOWN);
64  
65          // setup gpio pins #04, #05, #06 as an output pins and make sure they are all LOW at startup
66          GpioPin myLed[] =
67            { 
68              gpio.provisionOuputPin(Pin.GPIO_04, "LED #1", PinState.LOW),
69              gpio.provisionOuputPin(Pin.GPIO_05, "LED #2", PinState.LOW),
70              gpio.provisionOuputPin(Pin.GPIO_06, "LED #3", PinState.LOW) 
71            };
72  
73          // create a gpio control trigger on the input pin ; when the input goes HIGH, also set gpio pin #04 to HIGH
74          myButton.addTrigger(new GpioSetStateTrigger(PinState.HIGH, myLed[0], PinState.HIGH));
75  
76          // create a gpio control trigger on the input pin ; when the input goes LOW, also set gpio pin #04 to LOW
77          myButton.addTrigger(new GpioSetStateTrigger(PinState.LOW, myLed[0], PinState.LOW));
78  
79          // create a gpio synchronization trigger on the input pin; when the input changes, also set gpio pin #05 to same state
80          myButton.addTrigger(new GpioSyncStateTrigger(myLed[1]));
81  
82          // create a gpio pulse trigger on the input pin; when the input goes HIGH, also pulse gpio pin #06 to the HIGH state for 1 second
83          myButton.addTrigger(new GpioPulseStateTrigger(PinState.HIGH, myLed[2], 1000));
84  
85          // create a gpio callback trigger on gpio pin#4; when #4 changes state, perform a callback
86          // invocation on the user defined 'Callable' class instance
87          myButton.addTrigger(new GpioCallbackTrigger(new Callable<Void>()
88          {
89              public Void call() throws Exception
90              {
91                  System.out.println(" --> GPIO TRIGGER CALLBACK RECEIVED ");
92                  return null;
93              }
94          }));
95  
96          // keep program running until user aborts (CTRL-C)
97          for (;;)
98          {
99              Thread.sleep(500);
100         }
101     }
102 }
