4   import java.util.concurrent.Callable;

6   import com.pi4j.io.gpio.Gpio;
7   import com.pi4j.io.gpio.GpioFactory;
8   import com.pi4j.io.gpio.GpioPin;
9   import com.pi4j.io.gpio.GpioPinDirection;
10  import com.pi4j.io.gpio.GpioPinEdge;
11  import com.pi4j.io.gpio.GpioPinResistor;
12  import com.pi4j.io.gpio.GpioPinState;
13  import com.pi4j.io.gpio.trigger.GpioCallbackTrigger;
14  import com.pi4j.io.gpio.trigger.GpioPulseStateTrigger;
15  import com.pi4j.io.gpio.trigger.GpioSetStateTrigger;
16  import com.pi4j.io.gpio.trigger.GpioSyncStateTrigger;

18  /**
19   * This example code demonstrates how to setup simple triggers for GPIO pins on the Raspberry Pi.
20   * 
21   * @author Robert Savage
22   */
23  public class TriggerGpioExample
24  {
25      public static void main(String[] args) throws InterruptedException
26      {
27          // create gpio controller
28          Gpio gpio = GpioFactory.createInstance();
29  
30          // create a gpio control trigger on gpio pin#4; when #4 goes HIGH, also set pin #21 to HIGH
31          GpioSetStateTrigger triggerOn = new GpioSetStateTrigger(GpioPin.GPIO_04, GpioPinState.HIGH,
32                  GpioPin.GPIO_21, GpioPinState.HIGH);
33  
34          // create a gpio control trigger on gpio pin#4; when #4 goes LOW, also set pin #21 to LOW
35          GpioSetStateTrigger triggerOff = new GpioSetStateTrigger(GpioPin.GPIO_04, GpioPinState.LOW,
36                  GpioPin.GPIO_21, GpioPinState.LOW);
37  
38          // create a gpio synchronization trigger on gpio pin#4; when #4 changes, also set pin #22 to
39          // same state
40          GpioSyncStateTrigger triggerSync = new GpioSyncStateTrigger(GpioPin.GPIO_04,
41                  GpioPinState.allStates(), GpioPin.GPIO_22);
42  
43          // create a gpio pulse trigger on gpio pin#4; when #4 goes HIGH, also pulse pin #23 to the
44          // HIGH state for 1 second
45          GpioPulseStateTrigger triggerPulse = new GpioPulseStateTrigger(GpioPin.GPIO_04,
46                  GpioPinState.HIGH, GpioPin.GPIO_23, 1000);
47  
48          // create a gpio callback trigger on gpio pin#4; when #4 changes state, perform a callback
49          // invocation on the user defined 'Callable' class instance
50          GpioCallbackTrigger triggerCallback = new GpioCallbackTrigger(GpioPin.GPIO_04,
51                  GpioPinState.allStates(), new Callable<Void>()
52                  {
53                      public Void call() throws Exception
54                      {
55                          System.out.println(" --> GPIO TRIGGER CALLBACK RECEIVED ");
56                          return null;
57                      }
58                  });
59  
60          // register all the gpio triggers with the gpio controller
61          gpio.addTrigger(triggerOn);
62          gpio.addTrigger(triggerOff);
63          gpio.addTrigger(triggerSync);
64          gpio.addTrigger(triggerCallback);
65          gpio.addTrigger(triggerPulse);
66  
67          // setup gpio pin #4 as an input pin with its internal pull down resistor enabled
68          // (configure pin edge to both rising and falling to get notified for HIGH and LOW state
69          // changes)
70          gpio.setup(GpioPin.GPIO_04, GpioPinDirection.IN, GpioPinEdge.BOTH,
71                     GpioPinResistor.PULL_DOWN);
72  
73          // setup gpio pins #21,22,23 as an output pins and make sure they are all LOW at startup
74          GpioPin[] pins = { GpioPin.GPIO_21, GpioPin.GPIO_22, GpioPin.GPIO_23 };
75          gpio.setup(pins, GpioPinDirection.OUT);
76          gpio.setLow(pins);
77  
78          // keep program running until user aborts (CTRL-C)
79          for (;;)
80          {
81              Thread.sleep(500);
82          }
83      }
84  }
