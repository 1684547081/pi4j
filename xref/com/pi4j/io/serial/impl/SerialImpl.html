195     /**
196      * <p>
197      * This method is called to submit a string of data to the serial port transmit buffer.
198      * </p>
199      * 
200      * @param data <p>
201      *            A string of data to be transmitted.
202      *            </p>
203      */
204     public void write(String data)
205     {
206         // break data into packets of 1024 bytes
207         int position = 0;
208         while (position < data.length())
209         {
210             int length = 1024;
211             if (position + 1024 > data.length())
212                 com.pi4j.wiringpi.Serial.serialPuts(fileDescriptor, data.substring(position));
213             else
214                 com.pi4j.wiringpi.Serial.serialPuts(fileDescriptor,
215                                                     data.substring(position, (position + length)));
216             position += length;
217         }
218     }

288     /**
289      * <h1>Add Serial Event Listener</h1>
290      * 
291      * <p>
292      * Java consumer code can call this method to register itself as a listener for serial data
293      * events.
294      * </p>
295      * 
296      * @see #com.pi4j.io.serial.SerialDataListener
297      * @see #com.pi4j.io.serial.SerialDataEvent
298      * 
299      * @param listener <p>
300      *            A class instance that implements the SerialListener interface.
301      *            </p>
302      */
303     public synchronized void addListener(SerialDataListener listener)
304     {
305         // add the new listener to the list of listeners
306         listeners.addElement(listener);
307 
308         // if there is not a current listening monitor thread running,
309         // then lets start it now
310         if (monitor == null || monitor.isAlive() == false)
311         {
312             monitor = new SerialDataMonitorThread(this, listeners);
313             monitor.start();
314         }
315     }
316 
317     /**
318      * <h1>Remove Serial Event Listener</h1>
319      * 
320      * <p>
321      * Java consumer code can call this method to unregister itself as a listener for serial data
322      * events.
323      * </p>
324      * 
325      * @see #com.pi4j.io.serial.SerialDataListener
326      * @see #com.pi4j.io.serial.SerialDataEvent
327 
328      * 
329      * @param listener <p>
330      *            A class instance that implements the SerialListener interface.
331      *            </p>
332      */
333     public synchronized void removeListener(SerialDataListener listener)
334     {
335         // remove the listener from the list of listeners
336         listeners.removeElement(listener);
337 
338         // if there are not more listeners, then exit and destroy
339         // the monitor thread now
340         if (listeners.isEmpty() && monitor != null)
341         {
342             monitor.exit();
343             monitor = null;
344         }
345     }
