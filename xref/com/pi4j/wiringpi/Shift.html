19  /**
20   * <h1>WiringPi Shift Library</h1>
21   * 
22   * <p>
23   * WiringPi includes a shift library which more or less mimics the one in the Arduino system. This
24   * allows you to shift 8-bit data values out of the Pi, or into the Pi from devices such as
25   * shift-registers (e.g. 74,595) and so-on, although it can also be used in some bit-banging
26   * scenarios.
27   * </p>
28   * 
29   * <p>
30   * Before using the Pi4J library, you need to ensure that the Java VM in configured with access to
31   * the following system libraries:
32   * <ul>
33   * <li>pi4j</li>
34   * <li>wiringPi</li>
35   * </ul>
36   * <blockquote> This library depends on the wiringPi native system library.</br> (developed by
37   * Gordon Henderson @ <a href="https://projects.drogon.net/">https://projects.drogon.net/</a>)
38   * </blockquote>
39   * </p>
40   * 
41   * @see <a href="http://www.pi4j.com/">http://www.pi4j.com/</a>
42   * @see <a
43   *      href="https://projects.drogon.net/raspberry-pi/wiringpi/shift-library/">https://projects.drogon.net/raspberry-pi/wiringpi/shift-library/</a>
44   * @author Robert Savage (<a
45   *         href="http://www.savagehomeautomation.com">http://www.savagehomeautomation.com</a>)
46   */
47  public class Shift
48  {
49      public static final int LSBFIRST = 0;
50      public static final int MSBFIRST = 1;
51  
52      // Load the platform library
53      static
54      {
55          System.loadLibrary("pi4j");
56      }
57  
58      /**
59       * <h1>uint8_t shiftIn (uint8_t dPin, uint8_t cPin, uint8_t order);</h1>
60       * 
61       * <p>
62       * This shifts an 8-bit data value in with the data appearing on the dPin and the clock being
63       * sent out on the cPin. Order is either LSBFIRST or MSBFIRST. The data is sampled after the
64       * cPin goes high. (So cPin high, sample data, cPin low, repeat for 8 bits) The 8-bit value is
65       * returned by the function.
66       * </p>
67       * 
68       * @see <a
69       *      href="https://projects.drogon.net/raspberry-pi/wiringpi/shift-library/">https://projects.drogon.net/raspberry-pi/wiringpi/shift-library/</a>
70       * 
71       * @param dPin
72       * @param cPin
73       * @param order
74       * @return <p>
75       *         The 8-bit shifted value is returned by the function.
76       *         </p>
77       */
78      public static native byte shiftIn(byte dPin, byte cPin, byte order);
79  
80      /**
81       * <h1>void shiftOut (uint8_t dPin, uint8_t cPin, uint8_t order, uint8_t val);</h1>
82       * 
83       * <p>
84       * The shifts an 8-bit data value val out with the data being sent out on dPin and the clock
85       * being sent out on the cPin. order is as above. Data is clocked out on the rising or falling
86       * edge; ie. dPin is set, then cPin is taken high then low and repeated for the 8 bits.
87       * </p>
88       * 
89       * @see <a
90       *      href="https://projects.drogon.net/raspberry-pi/wiringpi/shift-library/">https://projects.drogon.net/raspberry-pi/wiringpi/shift-library/</a>
91       * 
92       * @param dPin
93       * @param cPin
94       * @param order
95       * @param val
96       */
97      public static native void shiftOut(byte dPin, byte cPin, byte order, byte val);
98  }
